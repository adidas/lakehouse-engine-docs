 19def _dry_run(bucket: str, object_paths: list) -> dict:
 20    """Build the dry run request return format.
 21
 22    Args:
 23        bucket: name of bucket to perform operation.
 24        object_paths: paths of object to list.
 25
 26    Returns:
 27        A dict with a list of objects that would be copied/deleted.
 28    """
 29    response = {}
 30
 31    for path in object_paths:
 32        if _check_directory(bucket, path):
 33            path = get_directory_path(path)
 34
 35        res = _list_objects_recursively(bucket=bucket, path=path)
 36
 37        if res:
 38            response[path] = res
 39        else:
 40            response[path] = ["No such key"]
 41
 42    return response

 45def _list_objects(
 46    s3_client: Any, bucket: str, path: str, paginator: str = ""
 47) -> Tuple[list, str]:
 48    """List 1000 objects in a bucket given a prefix and paginator in s3.
 49
 50    Args:
 51        bucket: name of bucket to perform the list.
 52        path: path to be used as a prefix.
 53        paginator: paginator token to be used.
 54
 55    Returns:
 56         A list of object names.
 57    """
 58    object_list = []
 59
 60    if not paginator:
 61        list_response = s3_client.list_objects_v2(Bucket=bucket, Prefix=path)
 62    else:
 63        list_response = s3_client.list_objects_v2(
 64            Bucket=bucket,
 65            Prefix=path,
 66            ContinuationToken=paginator,
 67        )
 68
 69    if FileManagerAPIKeys.CONTENTS.value in list_response:
 70        for obj in list_response[FileManagerAPIKeys.CONTENTS.value]:
 71            object_list.append(obj[FileManagerAPIKeys.KEY.value])
 72
 73    if FileManagerAPIKeys.CONTINUATION.value in list_response:
 74        pagination = list_response[FileManagerAPIKeys.CONTINUATION.value]
 75    else:
 76        pagination = ""
 77
 78    return object_list, pagination

 81def _list_objects_recursively(bucket: str, path: str) -> list:
 82    """Recursively list all objects given a prefix in s3.
 83
 84    Args:
 85        bucket: name of bucket to perform the list.
 86        path: path to be used as a prefix.
 87
 88    Returns:
 89        A list of object names fetched recursively.
 90    """
 91    object_list = []
 92    more_objects = True
 93    paginator = ""
 94
 95    s3 = boto3.client("s3")
 96
 97    while more_objects:
 98        temp_list, paginator = _list_objects(s3, bucket, path, paginator)
 99
100        object_list.extend(temp_list)
101
102        if not paginator:
103            more_objects = False
104
105    return object_list

108def _check_directory(bucket: str, path: str) -> bool:
109    """Checks if the object is a 'directory' in s3.
110
111    Args:
112        bucket: name of bucket to perform the check.
113        path: path to be used as a prefix.
114
115    Returns:
116        If path represents a 'directory'.
117    """
118    s3 = boto3.client("s3")
119    objects, _ = _list_objects(s3, bucket, path)
120    return len(objects) > 1

150        """Delete objects recursively in s3.
151
152        Params:

154            objects_paths: objects to be deleted.
155        """
156        s3 = boto3.client("s3")
157
158        for path in objects_paths:

164            more_objects = True
165            paginator = ""
166            objects_to_delete = []

172                for obj in objects_found:
173                    objects_to_delete.append({FileManagerAPIKeys.KEY.value: obj})
174
175                if not paginator:
176                    more_objects = False
177
178                response = s3.delete_objects(
179                    Bucket=bucket,
180                    Delete={FileManagerAPIKeys.OBJECTS.value: objects_to_delete},
181                )
182                self._logger.info(response)
183                objects_to_delete = []

308        source_bucket: str,
309        source_object: str,
310        destination_bucket: str,
311        destination_object: str,
312        dry_run: bool,
313    ) -> None:
314        """Copies objects and 'directories' in s3.
315
316        Args:

323        """
324        s3 = boto3.client("s3")
325
326        if dry_run:
327            response = _dry_run(bucket=source_bucket, object_paths=[source_object])
328
329            S3FileManager._logger.info("Paths that would be copied:")
330            S3FileManager._logger.info(response)
331        else:
332            original_object_name = source_object.split("/")[-1]
333
334            if _check_directory(source_bucket, source_object):
335                source_object = get_directory_path(source_object)
336
337                copy_object = _list_objects_recursively(
338                    bucket=source_bucket, path=source_object
339                )
340
341                for obj in copy_object:
342                    S3FileManager._logger.info(f"Copying obj: {obj}")
343
344                    final_path = obj.replace(source_object, "")
345
346                    response = s3.copy_object(
347                        Bucket=destination_bucket,
348                        CopySource={
349                            FileManagerAPIKeys.BUCKET.value: source_bucket,
350                            FileManagerAPIKeys.KEY.value: obj,
351                        },
352                        Key=f"{destination_object}/{original_object_name}/{final_path}",
353                    )
354                    S3FileManager._logger.info(response)
355            else:
356                S3FileManager._logger.info(f"Copying obj: {source_object}")
357
358                response = s3.copy_object(
359                    Bucket=destination_bucket,
360                    CopySource={
361                        FileManagerAPIKeys.BUCKET.value: source_bucket,
362                        FileManagerAPIKeys.KEY.value: source_object,
363                    },
364                    Key=f"""{destination_object}/{original_object_name}""",
365                )
366                S3FileManager._logger.info(response)

375    def _get_archived_object(bucket: str, object_key: str) -> Optional[Any]:
376        """Get the archived object if it's an object.
377
378        Args:
379            bucket: name of bucket to check get the object.
380            object_key: object to get.
381
382        Returns:
383            S3 Object if it's an archived object, otherwise None.
384        """
385        s3 = boto3.resource("s3")
386        object_to_restore = s3.Object(bucket, object_key)
387
388        if (
389            object_to_restore.storage_class is not None
390            and object_to_restore.storage_class in ARCHIVE_STORAGE_CLASS
391        ):
392            return object_to_restore
393        else:
394            return None

396    @staticmethod
397    def _check_object_restore_status(
398        bucket: str, object_key: str
399    ) -> Optional[RestoreStatus]:
400        """Check the restore status of the archive.
401
402        Args:
403            bucket: name of bucket to check the restore status.
404            object_key: object to check the restore status.
405
406        Returns:
407            The restore status represented by an enum, possible values are:
408                NOT_STARTED, ONGOING or RESTORED
409        """
410        archived_object = ArchiveFileManager._get_archived_object(bucket, object_key)
411
412        if archived_object is None:
413            status = None
414        elif archived_object.restore is None:
415            status = RestoreStatus.NOT_STARTED
416        elif 'ongoing-request="true"' in archived_object.restore:
417            status = RestoreStatus.ONGOING
418        else:
419            status = RestoreStatus.RESTORED
420
421        return status

423    @staticmethod
424    def check_restore_status(source_bucket: str, source_object: str) -> dict:
425        """Check the restore status of archived data.
426
427        Args:
428            source_bucket: name of bucket to check the restore status.
429            source_object: object to check the restore status.
430
431        Returns:
432            A dict containing the amount of objects in each status.
433        """
434        not_started_objects = 0
435        ongoing_objects = 0
436        restored_objects = 0
437        total_objects = 0
438
439        if _check_directory(source_bucket, source_object):
440            source_object = get_directory_path(source_object)
441
442        objects_to_restore = _list_objects_recursively(
443            bucket=source_bucket, path=source_object
444        )
445
446        for obj in objects_to_restore:
447            ArchiveFileManager._logger.info(f"Checking restore status for: {obj}")
448
449            restore_status = ArchiveFileManager._check_object_restore_status(
450                source_bucket, obj
451            )
452            if not restore_status:
453                ArchiveFileManager._logger.warning(
454                    f"Restore status not found for {source_bucket}/{obj}"
455                )
456            else:
457                total_objects += 1
458
459                if RestoreStatus.NOT_STARTED == restore_status:
460                    not_started_objects += 1
461                elif RestoreStatus.ONGOING == restore_status:
462                    ongoing_objects += 1
463                else:
464                    restored_objects += 1
465
466                ArchiveFileManager._logger.info(
467                    f"{obj} restore status is {restore_status.value}"
468                )
469
470        return {
471            "total_objects": total_objects,
472            "not_started_objects": not_started_objects,
473            "ongoing_objects": ongoing_objects,
474            "restored_objects": restored_objects,
475        }

477    @staticmethod
478    def _request_restore_object(
479        bucket: str, object_key: str, expiration: int, retrieval_tier: str
480    ) -> None:
481        """Request a restore of the archive.
482
483        Args:
484            bucket: name of bucket to perform the restore.
485            object_key: object to be restored.
486            expiration: restore expiration.
487            retrieval_tier: type of restore, possible values are:
488                Bulk, Standard or Expedited.
489        """
490        if not RestoreType.exists(retrieval_tier):
491            raise RestoreTypeNotFoundException(
492                f"Restore type {retrieval_tier} not supported."
493            )
494
495        if _check_directory(bucket, object_key):
496            object_key = get_directory_path(object_key)
497
498        archived_object = ArchiveFileManager._get_archived_object(bucket, object_key)
499
500        if archived_object and archived_object.restore is None:
501            ArchiveFileManager._logger.info(f"Restoring archive {bucket}/{object_key}.")
502            archived_object.restore_object(
503                RestoreRequest={
504                    "Days": expiration,
505                    "GlacierJobParameters": {"Tier": retrieval_tier},
506                }
507            )
508        else:
509            ArchiveFileManager._logger.info(
510                f"Restore request for {bucket}/{object_key} not performed."
511            )

590        )

595

603                    "Not all objects have been restored yet, checking the status again "

607                wait_time = 30
608                restore_status = ArchiveFileManager.check_restore_status(
609                    source_bucket, source_object
610                )

Request and wait for the restore to complete, polling the restore status.

Arguments:
  • source_bucket: name of bucket to perform the restore.
  • source_object: object to be restored.
  • restore_expiration: restore expiration in days.
  • retrieval_tier: type of restore, possible values are: Bulk, Standard or Expedited.
  • dry_run: if dry_run is set to True the function will print a dict with all the paths that would be deleted based on the given keys.
