 49    def execute(self) -> None:
 50        """Execute the Gold Asset Builder."""
 51        self._LOGGER.info(f"Reading {self.spec.lookup_table} as lkp_query_builder")
 52        lookup_query_builder_df = ExecEnv.SESSION.read.table(self.spec.lookup_table)
 53        ExecEnv.SESSION.read.table(self.spec.calendar_table).createOrReplaceTempView(
 54            "df_cal"
 55        )
 56        self._LOGGER.info(f"Generating calendar from {self.spec.calendar_table}")
 57
 58        query_label = self.spec.query_label_filter
 59        queue = self.spec.queue_filter
 60        cadence = self.spec.cadence_filter
 61
 62        self._LOGGER.info(f"Query Label Filter {query_label}")
 63        self._LOGGER.info(f"Queue Filter {queue}")
 64        self._LOGGER.info(f"Cadence Filter {cadence}")
 65
 66        gab_path = self.spec.gab_base_path
 67        self._LOGGER.info(f"Gab Base Path {gab_path}")
 68
 69        lookup_query_builder_df = lookup_query_builder_df.filter(
 70            (
 71                (lookup_query_builder_df.query_label.isin(query_label))
 72                & (lookup_query_builder_df.queue.isin(queue))
 73                & (lookup_query_builder_df.is_active != lit("N"))
 74            )
 75        )
 76
 77        lookup_query_builder_df.cache()
 78
 79        for use_case in lookup_query_builder_df.collect():
 80            self._process_use_case(
 81                use_case=use_case,
 82                lookup_query_builder=lookup_query_builder_df,
 83                selected_cadences=cadence,
 84                gab_path=gab_path,
 85            )
 86
 87        lookup_query_builder_df.unpersist()

 89    def _process_use_case(
 90        self,
 91        use_case: Row,
 92        lookup_query_builder: DataFrame,
 93        selected_cadences: list[str],
 94        gab_path: str,
 95    ) -> None:
 96        """Process each gab use case.
 97
 98        Args:
 99            use_case: gab use case to process.
100            lookup_query_builder: gab configuration data.
101            selected_cadences: selected cadences to process.
102            gab_path: gab base path used to get the use case stages sql files.
103        """
104        self._LOGGER.info(f"Executing use case: {use_case['query_label']}")
105
106        reconciliation = GABUtils.get_json_column_as_dict(
107            lookup_query_builder=lookup_query_builder,
108            query_id=use_case["query_id"],
109            query_column="recon_window",
110        )
111        self._LOGGER.info(f"reconcilation window - {reconciliation}")
112        configured_cadences = list(reconciliation.keys())
113
114        stages = GABUtils.get_json_column_as_dict(
115            lookup_query_builder=lookup_query_builder,
116            query_id=use_case["query_id"],
117            query_column="intermediate_stages",
118        )
119        self._LOGGER.info(f"intermediate stages - {stages}")
120
121        self._LOGGER.info(f"selected_cadences: {selected_cadences}")
122        self._LOGGER.info(f"configured_cadences: {configured_cadences}")
123        cadences = self._get_filtered_cadences(selected_cadences, configured_cadences)
124        self._LOGGER.info(f"filtered cadences - {cadences}")
125
126        latest_run_date, latest_config_date = self._get_latest_usecase_data(
127            use_case["query_id"]
128        )
129        self._LOGGER.info(f"latest_config_date: {latest_config_date}")
130        self._LOGGER.info(f"latest_run_date: - {latest_run_date}")
131        self._set_use_case_stage_template_file(stages, gab_path, use_case)
132        processed_cadences = []
133
134        for cadence in cadences:
135            is_cadence_processed = self._process_use_case_query_cadence(
136                cadence,
137                reconciliation,
138                use_case,
139                stages,
140                lookup_query_builder,
141            )
142            if is_cadence_processed:
143                processed_cadences.append(is_cadence_processed)
144
145        if processed_cadences:
146            self._generate_ddl(
147                latest_config_date=latest_config_date,
148                latest_run_date=latest_run_date,
149                query_id=use_case["query_id"],
150                lookup_query_builder=lookup_query_builder,
151            )
152        else:
153            self._LOGGER.info(
154                f"Skipping use case {use_case['query_label']}. No cadence processed "
155                "for the use case."
156            )

183    def _process_use_case_query_cadence(
184        self,
185        cadence: str,
186        reconciliation: dict,
187        use_case: Row,
188        stages: dict,
189        lookup_query_builder: DataFrame,
190    ) -> bool:
191        """Identify use case reconciliation window and cadence.
192
193        Args:
194            cadence:  cadence to process.
195            reconciliation: configured use case reconciliation window.
196            use_case: gab use case to process.
197            stages: use case stages with their configuration.
198            lookup_query_builder: gab configuration data.
199        """
200        selected_reconciliation_window = {}
201        selected_cadence = reconciliation.get(cadence)
202        self._LOGGER.info(f"Processing cadence: {cadence}")
203        self._LOGGER.info(f"Reconciliation Window - {selected_cadence}")
204
205        if selected_cadence:
206            selected_reconciliation_window = selected_cadence.get("recon_window")
207
208        self._LOGGER.info(f"{cadence}: {self.spec.start_date} - {self.spec.end_date}")
209
210        start_of_week = use_case["start_of_the_week"]
211
212        self._set_week_configuration_by_uc_start_of_week(start_of_week)
213
214        cadence_configuration_at_end_date = (
215            GABUtils.get_cadence_configuration_at_end_date(self.spec.end_date)
216        )
217
218        reconciliation_cadences = GABUtils().get_reconciliation_cadences(
219            cadence=cadence,
220            selected_reconciliation_window=selected_reconciliation_window,
221            cadence_configuration_at_end_date=cadence_configuration_at_end_date,
222            rerun_flag=self.spec.rerun_flag,
223        )
224
225        start_date_str = GABUtils.format_datetime_to_default(self.spec.start_date)
226        end_date_str = GABUtils.format_datetime_to_default(self.spec.end_date)
227
228        for reconciliation_cadence, snapshot_flag in reconciliation_cadences.items():
229            self._process_reconciliation_cadence(
230                reconciliation_cadence=reconciliation_cadence,
231                snapshot_flag=snapshot_flag,
232                cadence=cadence,
233                start_date_str=start_date_str,
234                end_date_str=end_date_str,
235                use_case=use_case,
236                lookup_query_builder=lookup_query_builder,
237                stages=stages,
238            )
239
240        return (cadence in reconciliation.keys()) or (
241            reconciliation_cadences is not None
242        )

244    def _process_reconciliation_cadence(
245        self,
246        reconciliation_cadence: str,
247        snapshot_flag: str,
248        cadence: str,
249        start_date_str: str,
250        end_date_str: str,
251        use_case: Row,
252        lookup_query_builder: DataFrame,
253        stages: dict,
254    ) -> None:
255        """Process use case reconciliation window.
256
257        Reconcile the pre-aggregated data to cover the late events.
258
259        Args:
260            reconciliation_cadence: reconciliation to process.
261            snapshot_flag: flag indicating if for this cadence the snapshot is enabled.
262            cadence: cadence to process.
263            start_date_str: start date of the period to process.
264            end_date_str: end date of the period to process.
265            use_case: gab use case to process.
266            lookup_query_builder: gab configuration data.
267            stages: use case stages with their configuration.
268
269        Example:
270            Cadence: week;
271            Reconciliation: monthly;
272            This means every weekend previous week aggregations will be calculated and
273                on month end we will reconcile the numbers calculated for last 4 weeks
274                to readjust the number for late events.
275        """
276        (
277            window_start_date,
278            window_end_date,
279            filter_start_date,
280            filter_end_date,
281        ) = GABCadenceManager().extended_window_calculator(
282            cadence,
283            reconciliation_cadence,
284            self.spec.current_date,
285            start_date_str,
286            end_date_str,
287            use_case["query_type"],
288            self.spec.rerun_flag,
289            snapshot_flag,
290        )
291
292        if use_case["timezone_offset"]:
293            filter_start_date = filter_start_date + timedelta(
294                hours=use_case["timezone_offset"]
295            )
296            filter_end_date = filter_end_date + timedelta(
297                hours=use_case["timezone_offset"]
298            )
299
300        filter_start_date_str = GABUtils.format_datetime_to_default(filter_start_date)
301        filter_end_date_str = GABUtils.format_datetime_to_default(filter_end_date)
302
303        partition_end = GABUtils.format_datetime_to_default(
304            (window_end_date - timedelta(days=1))
305        )
306
307        window_start_date_str = GABUtils.format_datetime_to_default(window_start_date)
308        window_end_date_str = GABUtils.format_datetime_to_default(window_end_date)
309
310        partition_filter = GABPartitionUtils.get_partition_condition(
311            filter_start_date_str, partition_end
312        )
313
314        self._LOGGER.info(
315            "extended window for start and end dates are: "
316            f"{filter_start_date_str} - {filter_end_date_str}"
317        )
318
319        unpersist_list = []
320
321        for i in range(1, len(stages) + 1):
322            stage = stages[str(i)]
323            templated_file = stage["templated_file"]
324            stage_file_path = stage["full_file_path"]
325
326            templated = self._process_use_case_query_step(
327                stage=stages[str(i)],
328                templated_file=templated_file,
329                use_case=use_case,
330                reconciliation_cadence=reconciliation_cadence,
331                cadence=cadence,
332                snapshot_flag=snapshot_flag,
333                window_start_date=window_start_date_str,
334                partition_end=partition_end,
335                filter_start_date=filter_start_date_str,
336                filter_end_date=filter_end_date_str,
337                partition_filter=partition_filter,
338            )
339
340            temp_stage_view_name = self._create_stage_view(
341                templated,
342                stages[str(i)],
343                window_start_date_str,
344                window_end_date_str,
345                use_case["query_id"],
346                use_case["query_label"],
347                cadence,
348                stage_file_path,
349            )
350            unpersist_list.append(temp_stage_view_name)
351
352        insert_success = self._generate_view_statement(
353            query_id=use_case["query_id"],
354            cadence=cadence,
355            temp_stage_view_name=temp_stage_view_name,
356            lookup_query_builder=lookup_query_builder,
357            window_start_date=window_start_date_str,
358            window_end_date=window_end_date_str,
359            query_label=use_case["query_label"],
360        )
361        self._LOGGER.info(f"Inserted data to generate the view: {insert_success}")
362
363        self._unpersist_cached_views(unpersist_list)

365    def _process_use_case_query_step(
366        self,
367        stage: dict,
368        templated_file: str,
369        use_case: Row,
370        reconciliation_cadence: str,
371        cadence: str,
372        snapshot_flag: str,
373        window_start_date: str,
374        partition_end: str,
375        filter_start_date: str,
376        filter_end_date: str,
377        partition_filter: str,
378    ) -> str:
379        """Process each use case step.
380
381        Process any intermediate view defined in the gab configuration table as step for
382            the use case.
383
384        Args:
385            stage: stage to process.
386            templated_file: sql file to process at this stage.
387            use_case: gab use case to process.
388            reconciliation_cadence: configured use case reconciliation window.
389            cadence: cadence to process.
390            snapshot_flag: flag indicating if for this cadence the snapshot is enabled.
391            window_start_date: start date for the configured stage.
392            partition_end: end date for the configured stage.
393            filter_start_date: filter start date to replace in the stage query.
394            filter_end_date: filter end date to replace in the stage query.
395            partition_filter: partition condition.
396        """
397        filter_col = stage["project_date_column"]
398        if stage["filter_date_column"]:
399            filter_col = stage["filter_date_column"]
400
401        # dummy value to avoid empty error if empty on the configuration
402        project_col = stage.get("project_date_column", "X")
403
404        gab_base_configuration_copy = copy.deepcopy(
405            GABCombinedConfiguration.COMBINED_CONFIGURATION.value
406        )
407
408        for item in gab_base_configuration_copy.values():
409            self._update_rendered_item_cadence(
410                reconciliation_cadence, cadence, project_col, item  # type: ignore
411            )
412
413        (
414            rendered_date,
415            rendered_to_date,
416            join_condition,
417        ) = self._get_cadence_configuration(
418            gab_base_configuration_copy,
419            cadence,
420            reconciliation_cadence,
421            snapshot_flag,
422            use_case["start_of_the_week"],
423            project_col,
424            window_start_date,
425            partition_end,
426        )
427
428        rendered_file = self._render_template_query(
429            templated=templated_file,
430            cadence=cadence,
431            start_of_the_week=use_case["start_of_the_week"],
432            query_id=use_case["query_id"],
433            rendered_date=rendered_date,
434            filter_start_date=filter_start_date,
435            filter_end_date=filter_end_date,
436            filter_col=filter_col,
437            timezone_offset=use_case["timezone_offset"],
438            join_condition=join_condition,
439            partition_filter=partition_filter,
440            rendered_to_date=rendered_to_date,
441        )
442
443        return rendered_file

478
479        Args:
480            query_id: use case query id.
481        """
482        last_success_run_sql = """
483            SELECT run_start_time
484            FROM {database}.gab_log_events
485            WHERE query_id = {query_id}
486            AND stage_name = 'Final Insert'
487            AND status = 'Success'
488            ORDER BY 1 DESC
489            LIMIT 1
490            """.format(  # nosec: B608
491            database=self.spec.target_database, query_id=query_id
492        )
493        try:
494            latest_run_date: datetime = ExecEnv.SESSION.sql(
495                last_success_run_sql
496            ).collect()[0][0]
497        except Exception:
498            latest_run_date = datetime.strptime(
499                "2020-01-01", GABDefaults.DATE_FORMAT.value
500            )

597    @classmethod
598    def _get_cadence_configuration(
599        cls,
600        use_case_configuration: dict,
601        cadence: str,
602        reconciliation_cadence: str,
603        snapshot_flag: str,
604        start_of_week: str,
605        project_col: str,
606        window_start_date: str,
607        partition_end: str,
608    ) -> tuple[str, str, str]:
609        """Get use case configuration fields to replace pre-configured parameters.
610
611        Args:
612            use_case_configuration: use case configuration.
613            cadence: cadence to process.
614            reconciliation_cadence: cadence to be reconciliated.
615            snapshot_flag: flag indicating if for this cadence the snapshot is enabled.
616            start_of_week: use case start of week (MONDAY or SUNDAY).
617            project_col: use case projection date column name.
618            window_start_date: start date for the configured stage.
619            partition_end: end date for the configured stage.
620
621        Returns:
622            rendered_from_date: projection start date.
623            rendered_to_date: projection end date.
624            join_condition: string containing the join condition to replace in the
625                templated query by jinja substitution.
626        """
627        cadence_dict = next(
628            (
629                dict(configuration)
630                for configuration in use_case_configuration.values()
631                if (
632                    (cadence in configuration["cadence"])
633                    and (reconciliation_cadence in configuration["recon"])
634                    and (snapshot_flag in configuration["snap_flag"])
635                    and (
636                        GABStartOfWeek.get_start_of_week()[start_of_week.upper()]
637                        in configuration["week_start"]
638                    )
639                )
640            ),
641            None,
642        )
643        rendered_from_date = None
644        rendered_to_date = None
645        join_condition = None
646
647        if cadence_dict:
648            rendered_from_date = (
649                cadence_dict[GABKeys.PROJECT_START]
650                .replace(GABReplaceableKeys.CADENCE, cadence)
651                .replace(GABReplaceableKeys.DATE_COLUMN, project_col)
652            )
653            rendered_to_date = (
654                cadence_dict[GABKeys.PROJECT_END]
655                .replace(GABReplaceableKeys.CADENCE, cadence)
656                .replace(GABReplaceableKeys.DATE_COLUMN, project_col)
657            )

598    @classmethod
599    def _get_cadence_configuration(
600        cls,
601        use_case_configuration: dict,
602        cadence: str,
603        reconciliation_cadence: str,
604        snapshot_flag: str,
605        start_of_week: str,
606        project_col: str,
607        window_start_date: str,
608        partition_end: str,
609    ) -> tuple[str, str, str]:
610        """Get use case configuration fields to replace pre-configured parameters.
611
612        Args:
613            use_case_configuration: use case configuration.
614            cadence: cadence to process.
615            reconciliation_cadence: cadence to be reconciliated.
616            snapshot_flag: flag indicating if for this cadence the snapshot is enabled.
617            start_of_week: use case start of week (MONDAY or SUNDAY).
618            project_col: use case projection date column name.
619            window_start_date: start date for the configured stage.
620            partition_end: end date for the configured stage.
621
622        Returns:
623            rendered_from_date: projection start date.
624            rendered_to_date: projection end date.
625            join_condition: string containing the join condition to replace in the
626                templated query by jinja substitution.
627        """
628        cadence_dict = next(
629            (
630                dict(configuration)
631                for configuration in use_case_configuration.values()
632                if (
633                    (cadence in configuration["cadence"])
634                    and (reconciliation_cadence in configuration["recon"])
635                    and (snapshot_flag in configuration["snap_flag"])
636                    and (
637                        GABStartOfWeek.get_start_of_week()[start_of_week.upper()]
638                        in configuration["week_start"]
639                    )
640                )
641            ),
642            None,
643        )
644        rendered_from_date = None
645        rendered_to_date = None
646        join_condition = None
647
648        if cadence_dict:
649            rendered_from_date = (
650                cadence_dict[GABKeys.PROJECT_START]
651                .replace(GABReplaceableKeys.CADENCE, cadence)
652                .replace(GABReplaceableKeys.DATE_COLUMN, project_col)
653            )
654            rendered_to_date = (
655                cadence_dict[GABKeys.PROJECT_END]
656                .replace(GABReplaceableKeys.CADENCE, cadence)
657                .replace(GABReplaceableKeys.DATE_COLUMN, project_col)
658            )
659
660            if cadence_dict[GABKeys.JOIN_SELECT]:
661                join_condition = """
662                 inner join (
663                     {join_select} from df_cal
664                     where calendar_date
665                     between '{bucket_start}' and '{bucket_end}'
666                 )
667                 df_cal on date({date_column})
668                     between df_cal.cadence_start_date and df_cal.cadence_end_date
669                 """.format(
670                    join_select=cadence_dict[GABKeys.JOIN_SELECT],
671                    bucket_start=window_start_date,
672                    bucket_end=partition_end,
673                    date_column=project_col,
674                )
675
676        return rendered_from_date, rendered_to_date, join_condition

726    def _create_stage_view(
727        self,
728        rendered_template: str,
729        stage: dict,
730        window_start_date: str,
731        window_end_date: str,
732        query_id: str,
733        query_label: str,
734        cadence: str,
735        stage_file_path: str,
736    ) -> str:
737        """Create each use case stage view.
738
739        Each stage has a specific order and refer to a specific SQL to be executed.
740
741        Args:
742            rendered_template: rendered stage SQL file.
743            stage: stage to process.
744            window_start_date: start date for the configured stage.
745            window_end_date: end date for the configured stage.
746            query_id: gab configuration table use case identifier.
747            query_label: gab configuration table use case name.
748            cadence: cadence to process.
749            stage_file_path: full stage file path (gab path + stage path).
750        """
751        run_start_time = datetime.now()
752        creation_status: str
753        error_message: Union[Exception, str]
754
755        try:
756            tmp = ExecEnv.SESSION.sql(rendered_template)
757            num_partitions = ExecEnv.SESSION.conf.get(
758                self._SPARK_DEFAULT_PARALLELISM_CONFIG,
759                self._SPARK_DEFAULT_PARALLELISM_VALUE,
760            )
761
762            if stage["repartition"]:
763                if stage["repartition"].get("numPartitions"):
764                    num_partitions = stage["repartition"]["numPartitions"]
765
766                if stage["repartition"].get("keys"):
767                    tmp = tmp.repartition(
768                        int(num_partitions), *stage["repartition"]["keys"]
769                    )
770                    self._LOGGER.info("Repartitioned on given Key(s)")
771                else:
772                    tmp = tmp.repartition(int(num_partitions))
773                    self._LOGGER.info("Repartitioned on given partition count")
774
775            temp_step_view_name: str = stage["table_alias"]
776            tmp.createOrReplaceTempView(temp_step_view_name)
777
778            if stage["storage_level"]:
779                ExecEnv.SESSION.sql(
780                    "CACHE TABLE {tbl} "
781                    "OPTIONS ('storageLevel' '{type}')".format(
782                        tbl=temp_step_view_name,
783                        type=stage["storage_level"],
784                    )
785                )
786                ExecEnv.SESSION.sql(
787                    "SELECT COUNT(*) FROM {tbl}".format(  # nosec: B608
788                        tbl=temp_step_view_name
789                    )
790                )
791                self._LOGGER.info(f"Cached stage view - {temp_step_view_name} ")
792
793            creation_status = "Success"
794            error_message = "NA"
795        except Exception as err:
796            creation_status = "Failed"
797            error_message = err
798            raise err
799        finally:
800            run_end_time = datetime.now()
801            GABUtils().logger(
802                run_start_time,
803                run_end_time,
804                window_start_date,
805                window_end_date,
806                query_id,
807                query_label,
808                cadence,
809                stage_file_path,
810                rendered_template,
811                creation_status,
812                error_message,
813                self.spec.target_database,
814            )
815
816        return temp_step_view_name
