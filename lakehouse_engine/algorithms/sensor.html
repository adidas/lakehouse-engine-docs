 23class Sensor(Algorithm):
 24    """Class representing a sensor to check if the upstream has new data."""
 25
 26    _LOGGER = LoggingHandler(__name__).get_logger()
 27
 28    def __init__(self, acon: dict):
 29        """Construct Sensor instances.
 30
 31        Args:
 32            acon: algorithm configuration.
 33        """
 34        self.spec: SensorSpec = SensorSpec.create_from_acon(acon=acon)
 35        self._validate_sensor_spec()
 36
 37        if self._check_if_sensor_already_exists():
 38            raise SensorAlreadyExistsException(
 39                "There's already a sensor registered with same id or assets!"
 40            )
 41
 42    def execute(self) -> bool:
 43        """Execute the sensor."""
 44        self._LOGGER.info(f"Starting {self.spec.input_spec.data_format} sensor...")
 45
 46        new_data_df = SensorUpstreamManager.read_new_data(sensor_spec=self.spec)
 47        if self.spec.input_spec.read_type == ReadType.STREAMING.value:
 48            Sensor._run_streaming_sensor(sensor_spec=self.spec, new_data_df=new_data_df)
 49        elif self.spec.input_spec.read_type == ReadType.BATCH.value:
 50            Sensor._run_batch_sensor(
 51                sensor_spec=self.spec,
 52                new_data_df=new_data_df,
 53            )
 54
 55        has_new_data = SensorControlTableManager.check_if_sensor_has_acquired_data(
 56            self.spec.sensor_id,
 57            self.spec.control_db_table_name,
 58        )
 59
 60        self._LOGGER.info(
 61            f"Sensor {self.spec.sensor_id} has previously "
 62            f"acquired data? {has_new_data}"
 63        )
 64
 65        if self.spec.fail_on_empty_result and not has_new_data:
 66            raise NoNewDataException(
 67                f"No data was acquired by {self.spec.sensor_id} sensor."
 68            )
 69
 70        return has_new_data
 71
 72    def _check_if_sensor_already_exists(self) -> bool:
 73        """Check if sensor already exists in the table to avoid duplicates."""
 74        row = SensorControlTableManager.read_sensor_table_data(
 75            sensor_id=self.spec.sensor_id,
 76            control_db_table_name=self.spec.control_db_table_name,
 77        )
 78
 79        if row and row.assets != self.spec.assets:
 80            return True
 81        else:
 82            row = SensorControlTableManager.read_sensor_table_data(
 83                assets=self.spec.assets,
 84                control_db_table_name=self.spec.control_db_table_name,
 85            )
 86            return row is not None and row.sensor_id != self.spec.sensor_id

 88    @classmethod
 89    def _run_streaming_sensor(
 90        cls, sensor_spec: SensorSpec, new_data_df: DataFrame
 91    ) -> None:
 92        """Run sensor in streaming mode (internally runs in batch mode)."""
 93
 94        def foreach_batch_check_new_data(df: DataFrame, batch_id: int) -> None:
 95            Sensor._run_batch_sensor(
 96                sensor_spec=sensor_spec,
 97                new_data_df=df,
 98            )
 99
100        new_data_df.writeStream.trigger(availableNow=True).option(
101            "checkpointLocation", sensor_spec.checkpoint_location
102        ).foreachBatch(foreach_batch_check_new_data).start().awaitTermination()
103
104    @classmethod
105    def _run_batch_sensor(
106        cls,
107        sensor_spec: SensorSpec,
108        new_data_df: DataFrame,
109    ) -> None:
110        """Run sensor in batch mode.
111
112        Args:
113            sensor_spec: sensor spec containing all sensor information.
114            new_data_df: DataFrame possibly containing new data.
115        """
116        new_data_first_row = SensorUpstreamManager.get_new_data(new_data_df)
117
118        cls._LOGGER.info(
119            f"Sensor {sensor_spec.sensor_id} has new data from upstream? "
120            f"{new_data_first_row is not None}"
121        )
122
123        if new_data_first_row:
124            SensorControlTableManager.update_sensor_status(
125                sensor_spec=sensor_spec,
126                status=SensorStatus.ACQUIRED_NEW_DATA.value,
127                upstream_key=(
128                    new_data_first_row.UPSTREAM_KEY
129                    if "UPSTREAM_KEY" in new_data_df.columns
130                    else None
131                ),
132                upstream_value=(
133                    new_data_first_row.UPSTREAM_VALUE
134                    if "UPSTREAM_VALUE" in new_data_df.columns
135                    else None
136                ),
137            )
138            cls._LOGGER.info(
139                f"Successfully updated sensor status for sensor "
140                f"{sensor_spec.sensor_id}..."
141            )
142
143    def _validate_sensor_spec(self) -> None:
144        """Validate if sensor spec Read Type is allowed for the selected Data Format."""
145        if InputFormat.exists(self.spec.input_spec.data_format):
146            if (
147                self.spec.input_spec.data_format
148                not in SENSOR_ALLOWED_DATA_FORMATS[self.spec.input_spec.read_type]
149            ):
150                raise NotImplementedError(
151                    f"A sensor has not been implemented yet for this data format or, "
152                    f"this data format is not available for the read_type"
153                    f" {self.spec.input_spec.read_type}. "
154                    f"Check the allowed combinations of read_type and data_formats:"
155                    f" {SENSOR_ALLOWED_DATA_FORMATS}"
156                )
157        else:
158            raise NotImplementedError(
159                f"Data format {self.spec.input_spec.data_format} isn't implemented yet."
160            )
