 61    @classmethod
 62    def run_dq_process(cls, dq_spec: DQSpec, data: DataFrame) -> DataFrame:
 63        """Run the specified data quality process on a dataframe.
 64
 65        Based on the dq_specs we apply the defined expectations on top of the dataframe
 66        in order to apply the necessary validations and then output the result of
 67        the data quality process.
 68
 69        Args:
 70            dq_spec: data quality specification.
 71            data: input dataframe to run the dq process on.
 72
 73        Returns:
 74            The DataFrame containing the results of the DQ process.
 75        """
 76        # import custom expectations for them to be available to be used.
 77        for expectation in DQDefaults.CUSTOM_EXPECTATION_LIST.value:
 78            importlib.__import__(
 79                "lakehouse_engine.dq_processors.custom_expectations." + expectation
 80            )
 81
 82        context = get_context(project_config=cls._get_data_context_config(dq_spec))
 83        context.add_datasource(**cls._get_data_source_defaults(dq_spec))
 84
 85        expectation_suite_name = (
 86            dq_spec.expectation_suite_name
 87            if dq_spec.expectation_suite_name
 88            else f"{dq_spec.spec_id}-{dq_spec.input_id}-{dq_spec.dq_type}"
 89        )
 90        context.add_or_update_expectation_suite(
 91            expectation_suite_name=expectation_suite_name
 92        )
 93
 94        batch_request = cls._get_batch_request(dq_spec, data)
 95
 96        # We only run the checkpoint for the Validator, as we don't want to allow
 97        # running this for all the expectations suggested by the Assistant.
 98        if dq_spec.dq_type == DQType.VALIDATOR.value:
 99            Validator.get_dq_validator(
100                context,
101                batch_request,
102                expectation_suite_name,
103                dq_spec.dq_functions,
104                dq_spec.critical_functions,
105            )
106
107            source_pk = cls._get_unexpected_rows_pk(dq_spec)
108            results, results_df = cls._configure_and_run_checkpoint(
109                dq_spec, context, batch_request, expectation_suite_name, source_pk
110            )
111
112            cls._write_to_result_sink(dq_spec, results_df)
113
114            cls._log_or_fail(results, dq_spec)
115
116            if (
117                dq_spec.tag_source_data
118                and dq_spec.result_sink_explode
119                and dq_spec.fail_on_error is not True
120            ):
121                data = Validator.tag_source_with_dq(source_pk, data, results_df)
122        elif dq_spec.dq_type == DQType.ASSISTANT.value:
123            Assistant.run_data_assistant(
124                context,
125                batch_request,
126                expectation_suite_name,
127                dq_spec.assistant_options or {},
128                data,
129                f"{cls._TIMESTAMP}_{dq_spec.spec_id}_{dq_spec.input_id}_profile",
130            )
131        else:
132            raise TypeError(
133                f"Type of Data Quality '{dq_spec.dq_type}' is not supported."
134            )
135
136        return data

139    def build_data_docs(
140        cls,
141        store_backend: str = DQDefaults.STORE_BACKEND.value,
142        local_fs_root_dir: str = None,
143        data_docs_local_fs: str = None,
144        data_docs_prefix: str = DQDefaults.DATA_DOCS_PREFIX.value,
145        bucket: str = None,
146        data_docs_bucket: str = None,
147        expectations_store_prefix: str = DQDefaults.EXPECTATIONS_STORE_PREFIX.value,
148        validations_store_prefix: str = DQDefaults.VALIDATIONS_STORE_PREFIX.value,
149        checkpoint_store_prefix: str = DQDefaults.CHECKPOINT_STORE_PREFIX.value,
150    ) -> None:
151        """Build Data Docs for the project.
152
153        This function does a full build of data docs based on all the great expectations
154        checkpoints in the specified location, getting all history of run/validations
155        executed and results.
156
157        Args:
158            store_backend: which store_backend to use (e.g. s3 or file_system).

165                (store DQ artefacts). Note: only applicable for store_backend s3.

172                Note: only applicable for store_backend s3.
173            checkpoint_store_prefix: prefix where to store checkpoints' data.
174                Note: only applicable for store_backend s3.
175        """
176        if store_backend == DQDefaults.STORE_BACKEND.value:
177            dq_spec = DQSpec(
178                spec_id="dq_validator",
179                input_id="dq",
180                dq_type=DQType.VALIDATOR.value,
181                store_backend=DQDefaults.STORE_BACKEND.value,
182                data_docs_prefix=data_docs_prefix,
183                bucket=bucket,
184                data_docs_bucket=data_docs_bucket,
185                expectations_store_prefix=expectations_store_prefix,
186                validations_store_prefix=validations_store_prefix,
187                checkpoint_store_prefix=checkpoint_store_prefix,
188            )
189        elif store_backend == DQDefaults.FILE_SYSTEM_STORE.value:
190            dq_spec = DQSpec(
191                spec_id="dq_validator",
192                input_id="dq",
193                dq_type=DQType.VALIDATOR.value,
194                store_backend=DQDefaults.FILE_SYSTEM_STORE.value,
195                local_fs_root_dir=local_fs_root_dir,
196                data_docs_local_fs=data_docs_local_fs,
197                data_docs_prefix=data_docs_prefix,
198            )
199        context = get_context(project_config=cls._get_data_context_config(dq_spec))
200        cls._LOGGER.info("The data docs were rebuilt")
201        context.build_data_docs()

203    @classmethod
204    def _check_critical_functions_tags(cls, failed_expectations: List[Any]) -> list:
205        critical_failure = []
206
207        for expectation in failed_expectations:
208            meta = expectation["meta"]
209            if meta and (
210                ("notes" in meta.keys() and "Critical function" in meta["notes"])
211                or (
212                    "content" in meta["notes"].keys()
213                    and "Critical function" in meta["notes"]["content"]
214                )
215            ):
216                critical_failure.append(expectation["expectation_type"])
217
218        return critical_failure

220    @classmethod
221    def _configure_and_run_checkpoint(
222        cls,
223        dq_spec: DQSpec,
224        context: EphemeralDataContext,
225        batch_request: RuntimeBatchRequest,
226        expectation_suite_name: str,
227        source_pk: List[str],
228    ) -> Tuple[CheckpointResult, DataFrame]:
229        """Configure, run and return checkpoint results.
230
231        A checkpoint is what enables us to run the validations of the expectations'
232        suite on the batches of data.
233
234        Args:
235            dq_spec: data quality specification.
236            context: the EphemeralDataContext containing the configurations for the data
237                source and store backend.
238            batch_request: run time batch request to be able to query underlying data.
239            expectation_suite_name: name of the expectation suite.
240            source_pk: the primary key of the source data.
241
242        Returns:
243            The checkpoint results in two types: CheckpointResult and Dataframe.
244        """
245        checkpoint_name = f"{dq_spec.spec_id}-{dq_spec.input_id}-checkpoint"
246        context.add_or_update_checkpoint(
247            name=checkpoint_name,
248            class_name=DQDefaults.DATA_CHECKPOINTS_CLASS_NAME.value,
249            config_version=DQDefaults.DATA_CHECKPOINTS_CONFIG_VERSION.value,
250            run_name_template=f"%Y%m%d-%H%M%S-{checkpoint_name}",
251        )
252
253        result_format: Dict[str, Any] = {
254            "result_format": dq_spec.gx_result_format,
255        }
256        if source_pk:
257            result_format = {
258                **result_format,
259                "unexpected_index_column_names": source_pk,
260            }
261
262        results = context.run_checkpoint(
263            checkpoint_name=checkpoint_name,
264            validations=[
265                {
266                    "batch_request": batch_request,
267                    "expectation_suite_name": expectation_suite_name,
268                }
269            ],
270            result_format=result_format,
271        )
272
273        return results, cls._transform_checkpoint_results(
274            results.to_json_dict(), dq_spec
275        )

277    @classmethod
278    def _explode_results(
279        cls,
280        df: DataFrame,
281        dq_spec: DQSpec,
282    ) -> DataFrame:
283        """Transform dq results dataframe exploding a set of columns.
284
285        Args:
286            df: dataframe with dq results to be exploded.
287            dq_spec: data quality specification.
288        """
289        df = df.withColumn(
290            "validation_results", explode("run_results.validation_result.results")
291        ).withColumn("source", lit(dq_spec.source))
292
293        new_columns = [
294            "validation_results.expectation_config.kwargs.*",
295            "run_results.validation_result.statistics.*",
296            "validation_results.expectation_config.expectation_type",
297            "validation_results.success as expectation_success",
298            "validation_results.exception_info",
299        ] + dq_spec.result_sink_extra_columns
300
301        df_exploded = df.selectExpr(*df.columns, *new_columns).drop(
302            *[c.replace(".*", "").split(" as")[0] for c in new_columns]
303        )
304
305        schema = df_exploded.schema.simpleString()
306        if "unexpected_index_list" in schema:
307            df_exploded = (
308                df_exploded.withColumn(
309                    "unexpected_index_list",
310                    array(struct(lit(True).alias("run_success"))),
311                )
312                if df.select(
313                    col("validation_results.result.unexpected_index_list")
314                ).dtypes[0][1]
315                == "array<string>"
316                else df_exploded.withColumn(
317                    "unexpected_index_list",
318                    transform(
319                        col("validation_results.result.unexpected_index_list"),
320                        lambda x: x.withField("run_success", lit(False)),
321                    ),
322                )
323            )
324
325        if "observed_value" in schema:
326            df_exploded = df_exploded.withColumn(
327                "observed_value", col("validation_results.result.observed_value")
328            )
329
330        return (
331            df_exploded.withColumn("run_time_year", year(to_timestamp("run_time")))
332            .withColumn("run_time_month", month(to_timestamp("run_time")))
333            .withColumn("run_time_day", dayofmonth(to_timestamp("run_time")))
334            .withColumn("checkpoint_config", to_json(col("checkpoint_config")))
335            .withColumn("run_results", to_json(col("run_results")))
336            .withColumn(
337                "kwargs", to_json(col("validation_results.expectation_config.kwargs"))
338            )
339            .withColumn("validation_results", to_json(col("validation_results")))
340        )

370    def _get_data_context_config(cls, dq_spec: DQSpec) -> DataContextConfig:
371        """Get the configuration of the data context.
372
373        Based on the configuration it is possible to define the backend to be
374        the file system (e.g. local file system) or S3, meaning that the DQ artefacts
375        will be stored according to this configuration.
376
377        Args:
378            dq_spec: data quality process specification.
379
380        Returns:
381            The DataContextConfig object configuration.
382        """
383        store_backend: Union[FilesystemStoreBackendDefaults, S3StoreBackendDefaults]
384        data_docs_site = None
385
386        if dq_spec.store_backend == DQDefaults.FILE_SYSTEM_STORE.value:
387            store_backend = FilesystemStoreBackendDefaults(
388                root_directory=dq_spec.local_fs_root_dir
389            )
390            data_docs_site = cls._get_data_docs_sites(
391                "local_site", store_backend.data_docs_sites, dq_spec
392            )
393        elif dq_spec.store_backend == DQDefaults.FILE_SYSTEM_S3_STORE.value:
394            store_backend = S3StoreBackendDefaults(
395                default_bucket_name=dq_spec.bucket,
396                validations_store_prefix=dq_spec.validations_store_prefix,
397                checkpoint_store_prefix=dq_spec.checkpoint_store_prefix,
398                expectations_store_prefix=dq_spec.expectations_store_prefix,
399                data_docs_prefix=dq_spec.data_docs_prefix,
400                data_docs_bucket_name=dq_spec.data_docs_bucket
401                if dq_spec.data_docs_bucket
402                else dq_spec.bucket,
403            )
404            data_docs_site = cls._get_data_docs_sites(
405                "s3_site", store_backend.data_docs_sites, dq_spec
406            )
407
408        return DataContextConfig(
409            store_backend_defaults=store_backend,
410            data_docs_sites=data_docs_site,
411            anonymous_usage_statistics=AnonymizedUsageStatisticsConfig(enabled=False),
412        )

475    @classmethod
476    def _get_failed_expectations(
477        cls, results: CheckpointResult, dq_spec: DQSpec
478    ) -> List[Any]:
479        """Get the failed expectations of a Checkpoint result.
480
481        Args:
482            results: the results of the DQ process.
483            dq_spec: data quality specification.
484
485        Returns: a list of failed expectations.
486        """
487        failed_expectations = []
488        for validation_result in results.list_validation_results():
489            expectations_results = validation_result["results"]
490            for result in expectations_results:
491                if not result["success"]:
492                    failed_expectations.append(result["expectation_config"])
493                    if result["exception_info"]["raised_exception"]:
494                        cls._LOGGER.error(
495                            f"""The expectation {str(result["expectation_config"])}
496                            raised the following exception:
497                            {result["exception_info"]["exception_message"]}"""
498                        )
499            cls._LOGGER.error(
500                f"{len(failed_expectations)} out of {len(expectations_results)} "
501                f"Data Quality Expectation(s) have failed! Failed Expectations: "
502                f"{failed_expectations}"
503            )
504
505            percentage_failure = 1 - (
506                validation_result["statistics"]["success_percent"] / 100
507            )
508
509            if (
510                dq_spec.max_percentage_failure
511                and dq_spec.max_percentage_failure < percentage_failure
512            ):
513                raise DQValidationsFailedException(
514                    f"Max error threshold is being surpassed! "
515                    f"Expected: {dq_spec.max_percentage_failure} "
516                    f"Got: {percentage_failure}"
517                )
518        return failed_expectations

543    @classmethod
544    def _log_or_fail(cls, results: CheckpointResult, dq_spec: DQSpec) -> None:
545        """Log the execution of the Data Quality process.
546
547        Args:
548            results: the results of the DQ process.
549            dq_spec: data quality specification.
550        """
551        if results["success"]:
552            cls._LOGGER.info(
553                "The data passed all the expectations defined. Everything looks good!"
554            )
555        else:
556            failed_expectations = cls._get_failed_expectations(results, dq_spec)
557            if dq_spec.critical_functions:
558                critical_failure = cls._check_critical_functions_tags(
559                    failed_expectations
560                )
561
562                if critical_failure:
563                    raise DQValidationsFailedException(
564                        f"Data Quality Validations Failed, the following critical "
565                        f"expectations failed: {critical_failure}."
566                    )
567            elif dq_spec.fail_on_error:
568                raise DQValidationsFailedException("Data Quality Validations Failed!")

570    @classmethod
571    def _transform_checkpoint_results(
572        cls, checkpoint_results: dict, dq_spec: DQSpec
573    ) -> DataFrame:
574        """Transforms the checkpoint results and creates new entries.
575
576        All the items of the dictionary are cast to a json like format.
577        The validation_result_identifier is extracted from the run_results column
578        into a separated column. All columns are cast to json like format.
579        After that the dictionary is converted into a dataframe.
580
581        Args:
582            checkpoint_results: dict with results of the checkpoint run.
583            dq_spec: data quality specification.
584
585        Returns:
586            Transformed results dataframe.
587        """
588        results_json_dict = loads(dumps(checkpoint_results))
589
590        results_dict = {}
591        for key, value in results_json_dict.items():
592            if key == "run_results":
593                checkpoint_result_identifier = list(value.keys())[0]
594                # check if the grabbed identifier is correct
595                if (
596                    str(checkpoint_result_identifier)
597                    .lower()
598                    .startswith(DQDefaults.VALIDATION_COLUMN_IDENTIFIER.value)
599                ):
600                    results_dict[
601                        "validation_result_identifier"
602                    ] = checkpoint_result_identifier
603                    results_dict["run_results"] = value[checkpoint_result_identifier]
604                else:
605                    raise DQCheckpointsResultsException(
606                        "The checkpoint result identifier format is not "
607                        "in accordance to what is expected"
608                    )
609            else:
610                results_dict[key] = value
611
612        df = ExecEnv.SESSION.createDataFrame(
613            [json.dumps(results_dict)], schema=StringType()
614        )
615        schema = schema_of_json(df.select("value").head()[0])
616        df = df.withColumn("value", from_json("value", schema)).select("value.*")
617
618        cols_to_expand = ["run_id"]
619        df = (
620            df.select(
621                [
622                    col(c) if c not in cols_to_expand else col(f"{c}.*")
623                    for c in df.columns
624                ]
625            )
626            .drop(*cols_to_expand)
627            .withColumn("spec_id", lit(dq_spec.spec_id))
628            .withColumn("input_id", lit(dq_spec.input_id))
629        )
630
631        return (
632            cls._explode_results(df, dq_spec)
633            if dq_spec.result_sink_explode
634            else df.withColumn(
635                "checkpoint_config", to_json(col("checkpoint_config"))
636            ).withColumn("run_results", to_json(col("run_results")))
637        )

639    @classmethod
640    def _write_to_result_sink(
641        cls,
642        dq_spec: DQSpec,
643        df: DataFrame,
644        data: OrderedDict = None,
645    ) -> None:
646        """Write dq results dataframe to a table or location.
647
648        It can be written:
649        - a raw output (having result_sink_explode set as False)

 62    @classmethod
 63    def run_dq_process(cls, dq_spec: DQSpec, data: DataFrame) -> DataFrame:
 64        """Run the specified data quality process on a dataframe.
 65
 66        Based on the dq_specs we apply the defined expectations on top of the dataframe
 67        in order to apply the necessary validations and then output the result of
 68        the data quality process.
 69
 70        Args:
 71            dq_spec: data quality specification.
 72            data: input dataframe to run the dq process on.
 73
 74        Returns:
 75            The DataFrame containing the results of the DQ process.
 76        """
 77        # import custom expectations for them to be available to be used.
 78        for expectation in DQDefaults.CUSTOM_EXPECTATION_LIST.value:
 79            importlib.__import__(
 80                "lakehouse_engine.dq_processors.custom_expectations." + expectation
 81            )
 82
 83        context = get_context(project_config=cls._get_data_context_config(dq_spec))
 84        context.add_datasource(**cls._get_data_source_defaults(dq_spec))
 85
 86        expectation_suite_name = (
 87            dq_spec.expectation_suite_name
 88            if dq_spec.expectation_suite_name
 89            else f"{dq_spec.spec_id}-{dq_spec.input_id}-{dq_spec.dq_type}"
 90        )
 91        context.add_or_update_expectation_suite(
 92            expectation_suite_name=expectation_suite_name
 93        )
 94
 95        batch_request = cls._get_batch_request(dq_spec, data)
 96
 97        # We only run the checkpoint for the Validator, as we don't want to allow
 98        # running this for all the expectations suggested by the Assistant.
 99        if dq_spec.dq_type == DQType.VALIDATOR.value:
100            Validator.get_dq_validator(
101                context,
102                batch_request,
103                expectation_suite_name,
104                dq_spec.dq_functions,
105                dq_spec.critical_functions,
106            )
107
108            source_pk = cls._get_unexpected_rows_pk(dq_spec)
109            results, results_df = cls._configure_and_run_checkpoint(
110                dq_spec, context, batch_request, expectation_suite_name, source_pk
111            )
112
113            cls._write_to_result_sink(dq_spec, results_df)
114
115            cls._log_or_fail(results, dq_spec)
116
117            if (
118                dq_spec.tag_source_data
119                and dq_spec.result_sink_explode
120                and dq_spec.fail_on_error is not True
121            ):
122                data = Validator.tag_source_with_dq(source_pk, data, results_df)
123        elif dq_spec.dq_type == DQType.ASSISTANT.value:
124            Assistant.run_data_assistant(
125                context,
126                batch_request,
127                expectation_suite_name,
128                dq_spec.assistant_options or {},
129                data,
130                f"{cls._TIMESTAMP}_{dq_spec.spec_id}_{dq_spec.input_id}_profile",
131            )
132        else:
133            raise TypeError(
134                f"Type of Data Quality '{dq_spec.dq_type}' is not supported."
135            )
136
137        return data

222    def _configure_and_run_checkpoint(
223        cls,
224        dq_spec: DQSpec,
225        context: EphemeralDataContext,
226        batch_request: RuntimeBatchRequest,
227        expectation_suite_name: str,
228        source_pk: List[str],

234

236            dq_spec: data quality specification.
237            context: the EphemeralDataContext containing the configurations for the data

252        )
253

255            "result_format": dq_spec.gx_result_format,
256        }

279    def _explode_results(
280        cls,
281        df: DataFrame,
282        dq_spec: DQSpec,
283    ) -> DataFrame:
284        """Transform dq results dataframe exploding a set of columns.
285
286        Args:
287            df: dataframe with dq results to be exploded.
288            dq_spec: data quality specification.
289        """
290        df = df.withColumn(

299            "validation_results.exception_info",
300        ] + dq_spec.result_sink_extra_columns
301
302        df_exploded = df.selectExpr(*df.columns, *new_columns).drop(
303            *[c.replace(".*", "").split(" as")[0] for c in new_columns]
304        )
305
306        schema = df_exploded.schema.simpleString()
307        if "unexpected_index_list" in schema:
308            df_exploded = (
309                df_exploded.withColumn(

311                    array(struct(lit(True).alias("run_success"))),

314                    col("validation_results.result.unexpected_index_list")

318                    "unexpected_index_list",
319                    transform(
320                        col("validation_results.result.unexpected_index_list"),

322                    ),
323                )
324            )
325
326        if "observed_value" in schema:
327            df_exploded = df_exploded.withColumn(
328                "observed_value", col("validation_results.result.observed_value")
329            )
330
331        return (
332            df_exploded.withColumn("run_time_year", year(to_timestamp("run_time")))
333            .withColumn("run_time_month", month(to_timestamp("run_time")))
334            .withColumn("run_time_day", dayofmonth(to_timestamp("run_time")))

338                "kwargs", to_json(col("validation_results.expectation_config.kwargs"))

476    @classmethod
477    def _get_failed_expectations(
478        cls, results: CheckpointResult, dq_spec: DQSpec
479    ) -> List[Any]:
480        """Get the failed expectations of a Checkpoint result.
481
482        Args:
483            results: the results of the DQ process.
484            dq_spec: data quality specification.
485
486        Returns: a list of failed expectations.
487        """
488        failed_expectations = []
489        for validation_result in results.list_validation_results():
490            expectations_results = validation_result["results"]
491            for result in expectations_results:
492                if not result["success"]:
493                    failed_expectations.append(result["expectation_config"])
494                    if result["exception_info"]["raised_exception"]:
495                        cls._LOGGER.error(
496                            f"""The expectation {str(result["expectation_config"])}
497                            raised the following exception:
498                            {result["exception_info"]["exception_message"]}"""
499                        )
500            cls._LOGGER.error(
501                f"{len(failed_expectations)} out of {len(expectations_results)} "
502                f"Data Quality Expectation(s) have failed! Failed Expectations: "
503                f"{failed_expectations}"
504            )
505
506            percentage_failure = 1 - (
507                validation_result["statistics"]["success_percent"] / 100
508            )
509
510            if (
511                dq_spec.max_percentage_failure
512                and dq_spec.max_percentage_failure < percentage_failure
513            ):
514                raise DQValidationsFailedException(
515                    f"Max error threshold is being surpassed! "
516                    f"Expected: {dq_spec.max_percentage_failure} "
517                    f"Got: {percentage_failure}"
518                )
519        return failed_expectations

571    @classmethod
572    def _transform_checkpoint_results(
573        cls, checkpoint_results: dict, dq_spec: DQSpec
574    ) -> DataFrame:
575        """Transforms the checkpoint results and creates new entries.
576
577        All the items of the dictionary are cast to a json like format.
578        The validation_result_identifier is extracted from the run_results column
579        into a separated column. All columns are cast to json like format.
580        After that the dictionary is converted into a dataframe.
581
582        Args:
583            checkpoint_results: dict with results of the checkpoint run.
584            dq_spec: data quality specification.
585
586        Returns:
587            Transformed results dataframe.
588        """
589        results_json_dict = loads(dumps(checkpoint_results))
590
591        results_dict = {}
592        for key, value in results_json_dict.items():
593            if key == "run_results":
594                checkpoint_result_identifier = list(value.keys())[0]
595                # check if the grabbed identifier is correct
596                if (
597                    str(checkpoint_result_identifier)
598                    .lower()
599                    .startswith(DQDefaults.VALIDATION_COLUMN_IDENTIFIER.value)
600                ):
601                    results_dict[
602                        "validation_result_identifier"
603                    ] = checkpoint_result_identifier
604                    results_dict["run_results"] = value[checkpoint_result_identifier]
605                else:
606                    raise DQCheckpointsResultsException(
607                        "The checkpoint result identifier format is not "
608                        "in accordance to what is expected"
609                    )
610            else:
611                results_dict[key] = value
612
613        df = ExecEnv.SESSION.createDataFrame(
614            [json.dumps(results_dict)], schema=StringType()
615        )
616        schema = schema_of_json(df.select("value").head()[0])
617        df = df.withColumn("value", from_json("value", schema)).select("value.*")
618
619        cols_to_expand = ["run_id"]
620        df = (
621            df.select(
622                [
623                    col(c) if c not in cols_to_expand else col(f"{c}.*")
624                    for c in df.columns
625                ]
626            )
627            .drop(*cols_to_expand)
628            .withColumn("spec_id", lit(dq_spec.spec_id))
629            .withColumn("input_id", lit(dq_spec.input_id))
630        )
631
632        return (
633            cls._explode_results(df, dq_spec)
634            if dq_spec.result_sink_explode
635            else df.withColumn(
636                "checkpoint_config", to_json(col("checkpoint_config"))
637            ).withColumn("run_results", to_json(col("run_results")))
638        )

640    @classmethod

643        dq_spec: DQSpec,
644        df: DataFrame,
645        data: OrderedDict = None,
646    ) -> None:
647        """Write dq results dataframe to a table or location.
648
649        It can be written:
650        - a raw output (having result_sink_explode set as False)
651        - an exploded output (having result_sink_explode set as True), which
652        is more prepared for analysis, with some columns exploded, flatten and
653        transformed. It can also be set result_sink_extra_columns with other
654        columns desired to have in the output table or location.
655
656        Args:

658            df: dataframe with dq results to write.
659            data: list of all dfs generated on previous steps before writer.
660        """
661        if dq_spec.result_sink_db_table or dq_spec.result_sink_location:
662            options = {"mergeSchema": "true"} if dq_spec.result_sink_explode else {}
663

 75            The DataFrame containing the results of the DQ process.
 76        """
 77        # import custom expectations for them to be available to be used.
 78        for expectation in DQDefaults.CUSTOM_EXPECTATION_LIST.value:
 79            importlib.__import__(
 80                "lakehouse_engine.dq_processors.custom_expectations." + expectation
 81            )
 82
 83        context = get_context(project_config=cls._get_data_context_config(dq_spec))
 84        context.add_datasource(**cls._get_data_source_defaults(dq_spec))
 85
 86        expectation_suite_name = (
 87            dq_spec.expectation_suite_name
 88            if dq_spec.expectation_suite_name
 89            else f"{dq_spec.spec_id}-{dq_spec.input_id}-{dq_spec.dq_type}"
 90        )
 91        context.add_or_update_expectation_suite(
 92            expectation_suite_name=expectation_suite_name
 93        )
 94

 96
 97        # We only run the checkpoint for the Validator, as we don't want to allow
 98        # running this for all the expectations suggested by the Assistant.
 99        if dq_spec.dq_type == DQType.VALIDATOR.value:
100            Validator.get_dq_validator(
101                context,
102                batch_request,
103                expectation_suite_name,
104                dq_spec.dq_functions,
105                dq_spec.critical_functions,
106            )
107
108            source_pk = cls._get_unexpected_rows_pk(dq_spec)
109            results, results_df = cls._configure_and_run_checkpoint(
110                dq_spec, context, batch_request, expectation_suite_name, source_pk
111            )
112
113            cls._write_to_result_sink(dq_spec, results_df)
114
115            cls._log_or_fail(results, dq_spec)
116
117            if (
118                dq_spec.tag_source_data
119                and dq_spec.result_sink_explode
120                and dq_spec.fail_on_error is not True
121            ):
122                data = Validator.tag_source_with_dq(source_pk, data, results_df)
123        elif dq_spec.dq_type == DQType.ASSISTANT.value:

125                context,

131            )
132        else:
133            raise TypeError(
134                f"Type of Data Quality '{dq_spec.dq_type}' is not supported."
135            )
