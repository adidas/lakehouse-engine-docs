 13    def __init__(self, output_spec: OutputSpec, df: DataFrame, data: OrderedDict):
 14        """Construct FileWriter instances.
 15
 16        Args:
 17            output_spec: output specification
 18            df: dataframe to be written.
 19            data: list of all dfs generated on previous steps before writer.
 20        """
 21        super().__init__(output_spec, df, data)
 22
 23    def write(self) -> None:
 24        """Write data to files."""
 25        if not self._df.isStreaming:
 26            self._write_to_files_in_batch_mode(self._df, self._output_spec)
 27        else:
 28            self._write_to_files_in_streaming_mode(
 29                self._df, self._output_spec, self._data
 30            )
 31
 32    @staticmethod
 33    def _write_to_files_in_batch_mode(df: DataFrame, output_spec: OutputSpec) -> None:
 34        """Write to files in batch mode.
 35
 36        Args:
 37            df: dataframe to write.
 38            output_spec: output specification.
 39        """
 40        df.write.format(output_spec.data_format).partitionBy(
 41            output_spec.partitions
 42        ).options(**output_spec.options if output_spec.options else {}).mode(
 43            output_spec.write_type
 44        ).save(
 45            output_spec.location
 46        )
 47
 48    @staticmethod
 49    def _write_to_files_in_streaming_mode(
 50        df: DataFrame, output_spec: OutputSpec, data: OrderedDict
 51    ) -> None:
 52        """Write to files in streaming mode.
 53
 54        Args:
 55            df: dataframe to write.
 56            output_spec: output specification.
 57            data: list of all dfs generated on previous steps before writer.
 58        """
 59        df_writer = df.writeStream.trigger(**Writer.get_streaming_trigger(output_spec))
 60
 61        if (
 62            output_spec.streaming_micro_batch_transformers
 63            or output_spec.streaming_micro_batch_dq_processors
 64        ):
 65            stream_df = (
 66                df_writer.options(**output_spec.options if output_spec.options else {})
 67                .foreachBatch(
 68                    FileWriter._write_transformed_micro_batch(output_spec, data)
 69                )
 70                .start()
 71            )
 72        else:
 73            stream_df = (
 74                df_writer.format(output_spec.data_format)
 75                .partitionBy(output_spec.partitions)
 76                .options(**output_spec.options if output_spec.options else {})
 77                .outputMode(output_spec.write_type)
 78                .start(output_spec.location)
 79            )
 80
 81        if output_spec.streaming_await_termination:
 82            stream_df.awaitTermination(output_spec.streaming_await_termination_timeout)
 83
 84    @staticmethod
 85    def _write_transformed_micro_batch(  # type: ignore
 86        output_spec: OutputSpec, data: OrderedDict
 87    ) -> Callable:
 88        """Define how to write a streaming micro batch after transforming it.
 89
 90        Args:
 91            output_spec: output specification.
 92            data: list of all dfs generated on previous steps before writer.
 93
 94        Returns:
 95            A function to be executed in the foreachBatch spark write method.
 96        """
 97
 98        def inner(batch_df: DataFrame, batch_id: int) -> None:
 99            transformed_df = Writer.get_transformed_micro_batch(
100                output_spec, batch_df, batch_id, data
101            )
102
103            if output_spec.streaming_micro_batch_dq_processors:
104                transformed_df = Writer.run_micro_batch_dq_process(
105                    transformed_df, output_spec.streaming_micro_batch_dq_processors
106                )
107
108            FileWriter._write_to_files_in_batch_mode(transformed_df, output_spec)
109
110        return inner
