37    @classmethod
38    def _get_writer_name(cls, spec: OutputSpec) -> str:
39        """Get the writer name according to the output specification.
40
41        Args:
42            OutputSpec spec: output specification to write data.
43
44        Returns:
45            Writer: writer name that will be created to write the data.
46        """
47        if spec.db_table and spec.write_type != WriteType.MERGE.value:
48            writer_name = OutputFormat.TABLE.value
49        elif (
50            spec.data_format == OutputFormat.DELTAFILES.value or spec.db_table
51        ) and spec.write_type == WriteType.MERGE.value:
52            writer_name = OutputFormat.DELTAFILES.value
53        elif spec.data_format in FILE_OUTPUT_FORMATS:
54            writer_name = OutputFormat.FILE.value
55        else:
56            writer_name = spec.data_format
57        return writer_name
