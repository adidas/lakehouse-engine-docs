 58    @classmethod
 59    def get_transformed_micro_batch(
 60        cls,
 61        output_spec: OutputSpec,
 62        batch_df: DataFrame,
 63        batch_id: int,
 64        data: OrderedDict,
 65    ) -> DataFrame:
 66        """Get the result of the transformations applied to a micro batch dataframe.
 67
 68        Args:
 69            output_spec: output specification associated with the writer.
 70            batch_df: batch dataframe (given from streaming foreachBatch).
 71            batch_id: if of the batch (given from streaming foreachBatch).
 72            data: list of all dfs generated on previous steps before writer
 73                to be available on micro batch transforms.
 74
 75        Returns:
 76            The transformed dataframe.
 77        """
 78        transformed_df = batch_df
 79        if output_spec.with_batch_id:
 80            transformed_df = transformed_df.withColumn("lhe_batch_id", lit(batch_id))
 81
 82        for transformer in output_spec.streaming_micro_batch_transformers:
 83            transformed_df = transformed_df.transform(
 84                TransformerFactory.get_transformer(transformer, data)
 85            )
 86
 87        return transformed_df

 89    @classmethod
 90    def get_streaming_trigger(cls, output_spec: OutputSpec) -> Dict:
 91        """Define which streaming trigger will be used.
 92
 93        Args:
 94            output_spec: output specification.
 95
 96        Returns:
 97            A dict containing streaming trigger.
 98        """
 99        trigger: Dict[str, Any] = {}
100
101        if output_spec.streaming_available_now:
102            trigger["availableNow"] = output_spec.streaming_available_now
103        elif output_spec.streaming_once:
104            trigger["once"] = output_spec.streaming_once
105        elif output_spec.streaming_processing_time:
106            trigger["processingTime"] = output_spec.streaming_processing_time
107        elif output_spec.streaming_continuous:
108            trigger["continuous"] = output_spec.streaming_continuous
109        else:
110            raise NotImplementedError(
111                "The requested output spec streaming trigger is not supported."
112            )
113
114        return trigger
