 17@dataclass
 18class SAPBWExtraction(JDBCExtraction):
 19    """Configurations available for an Extraction from SAP BW.
 20
 21    It inherits from SAPBWExtraction configurations, so it can use
 22    and/or overwrite those configurations.
 23
 24    These configurations cover:
 25    - latest_timestamp_input_col: the column containing the actrequest timestamp
 26        in the dataset in latest_timestamp_data_location. Default:
 27        "actrequest_timestamp".
 28    - act_request_table: the name of the SAP BW activation requests table.
 29        Composed of database.table. Default: SAPPHA.RSODSACTREQ.
 30    - request_col_name: name of the column having the request to join
 31        with the activation request table. Default: actrequest.
 32    - act_req_join_condition: the join condition into activation table
 33        can be changed using this property.
 34        Default: 'changelog_tbl.request = act_req.request_col_name'.
 35    - odsobject: name of BW Object, used for joining with the activation request
 36        table to get the max actrequest_timestamp to consider while filtering
 37        the changelog table.
 38    - include_changelog_tech_cols: whether to include the technical columns
 39        (usually coming from the changelog) table or not. Default: True.
 40    - extra_cols_act_request: list of columns to be added from act request table.
 41        It needs to contain the prefix "act_req.". E.g. "act_req.col1
 42        as column_one, act_req.col2 as column_two".
 43    - get_timestamp_from_act_request: whether to get init timestamp
 44        from act request table or assume current/given timestamp.
 45    - sap_bw_schema: sap bw schema. Default: SAPPHA.
 46    - max_timestamp_custom_schema: the custom schema to apply on the calculation of
 47        the max timestamp to consider for the delta extractions.
 48        Default: timestamp DECIMAL(23,0).
 49    - default_max_timestamp: the timestamp to use as default, when it is not possible
 50        to derive one.
 51    """
 52
 53    latest_timestamp_input_col: str = "actrequest_timestamp"
 54    act_request_table: str = "SAPPHA.RSODSACTREQ"
 55    request_col_name: str = "actrequest"
 56    act_req_join_condition: Optional[str] = None
 57    odsobject: Optional[str] = None
 58    include_changelog_tech_cols: bool = True
 59    extra_cols_act_request: Optional[str] = None
 60    get_timestamp_from_act_request: bool = False
 61    sap_bw_schema: str = "SAPPHA"
 62    max_timestamp_custom_schema: str = "timestamp DECIMAL(15,0)"
 63    default_max_timestamp: str = "197000000000000"

 70        """Construct SAPBWExtractionUtils.

 78        self._MAX_TIMESTAMP_QUERY = f""" --# nosec
 79                (SELECT COALESCE(MAX(timestamp),
 80                    {self._BW_EXTRACTION.default_max_timestamp}) as timestamp
 81                FROM {self._BW_EXTRACTION.act_request_table}

 87    def get_changelog_table(self) -> str:
 88        """Get the changelog table, given an odsobject.
 89
 90        Returns:
 91             String to use as changelog_table.
 92        """
 93        if (
 94            self._BW_EXTRACTION.odsobject is not None
 95            and self._BW_EXTRACTION.changelog_table is None
 96            and self._BW_EXTRACTION.extraction_type != JDBCExtractionType.INIT.value
 97        ):
 98            escaped_odsobject = copy(self._BW_EXTRACTION.odsobject).replace("_", "$_")
 99
100            if self._BW_EXTRACTION.sap_bw_schema:
101                system_table = f"{self._BW_EXTRACTION.sap_bw_schema}.RSTSODS"
102            else:
103                system_table = "RSTSODS"
104
105            jdbc_args = {
106                "url": self._BW_EXTRACTION.url,
107                "table": f""" -- # nosec
108                    (SELECT ODSNAME_TECH
109                    FROM {system_table}
110                    WHERE ODSNAME LIKE '8{escaped_odsobject}$_%'
111                        ESCAPE '$' AND USERAPP = 'CHANGELOG' AND VERSION = '000')
112                """,  # nosec: B608
113                "properties": {
114                    "user": self._BW_EXTRACTION.user,
115                    "password": self._BW_EXTRACTION.password,
116                    "driver": self._BW_EXTRACTION.driver,
117                },
118            }
119            from lakehouse_engine.io.reader_factory import ReaderFactory
120
121            changelog_df = ReaderFactory.get_data(
122                InputSpec(
123                    spec_id="changelog_table",
124                    data_format=InputFormat.JDBC.value,
125                    read_type=ReadType.BATCH.value,
126                    jdbc_args=jdbc_args,
127                )
128            )
129            changelog_table = (
130                f'{self._BW_EXTRACTION.sap_bw_schema}."{changelog_df.first()[0]}"'
131                if self._BW_EXTRACTION.sap_bw_schema
132                else str(changelog_df.first()[0])
133            )
134        else:
135            changelog_table = (
136                self._BW_EXTRACTION.changelog_table
137                if self._BW_EXTRACTION.changelog_table
138                else f"{self._BW_EXTRACTION.dbtable}_cl"
139            )
140        self._LOGGER.info(f"The changelog table derived is: '{changelog_table}'")
141
142        return changelog_table

164    def _get_init_query(self) -> Tuple[str, str]:
165        """Get a query to do an init load based on a DSO on a SAP BW system.
166
167        Returns:
168            A query to submit to SAP BW for the initial data extraction. The query
169            is enclosed in parentheses so that Spark treats it as a table and supports
170            it in the dbtable option.
171        """
172        if self._BW_EXTRACTION.get_timestamp_from_act_request:
173            # check if we are dealing with a DSO of type Write Optimised
174            if self._BW_EXTRACTION.dbtable == self._BW_EXTRACTION.changelog_table:
175                extraction_query = self._get_init_extraction_query_act_req_timestamp()
176            else:
177                raise AttributeError(
178                    "Not able to get the extraction query. The option "
179                    "'get_timestamp_from_act_request' is only "
180                    "available/useful for DSOs of type Write Optimised."
181                )
182        else:
183            extraction_query = self._get_init_extraction_query()
184
185        predicates_query = f"""
186        (SELECT DISTINCT({self._BW_EXTRACTION.partition_column})
187        FROM {self._BW_EXTRACTION.dbtable} t)
188        """  # nosec: B608
189
190        return extraction_query, predicates_query

192    def _get_init_extraction_query(self) -> str:
193        """Get extraction query based on given/current timestamp.
194
195        Returns:
196            A query to submit to SAP BW for the initial data extraction.
197        """
198        changelog_tech_cols = (
199            f"""'0' AS request,
200                CAST({self._BW_EXTRACTION.extraction_timestamp} AS DECIMAL(15, 0))
201                 AS actrequest_timestamp,
202                '0' AS datapakid,
203                0 AS partno,
204                0 AS record,"""
205            if self._BW_EXTRACTION.include_changelog_tech_cols
206            else f"CAST({self._BW_EXTRACTION.extraction_timestamp} "
207            f"AS DECIMAL(15, 0))"
208            f" AS actrequest_timestamp,"
209        )
210
211        extraction_query = f"""
212                (SELECT t.*,
213                    {changelog_tech_cols}
214                    CAST({self._BW_EXTRACTION.extraction_timestamp}
215                        AS DECIMAL(15, 0)) AS extraction_start_timestamp
216                FROM {self._BW_EXTRACTION.dbtable} t
217                )"""  # nosec: B608
218
219        return extraction_query

221    def _get_init_extraction_query_act_req_timestamp(self) -> str:
222        """Get extraction query assuming the init timestamp from act_request table.
223
224        Returns:
225            A query to submit to SAP BW for the initial data extraction from
226            write optimised DSOs, receiving the actrequest_timestamp from
227            the activation requests table.
228        """
229        extraction_query = f"""
230            (SELECT t.*,
231                act_req.timestamp as actrequest_timestamp,
232                CAST({self._BW_EXTRACTION.extraction_timestamp} AS DECIMAL(15, 0))
233                 AS extraction_start_timestamp
234            FROM {self._BW_EXTRACTION.dbtable} t
235            JOIN {self._BW_EXTRACTION.act_request_table} AS act_req ON
236                t.request = act_req.{self._BW_EXTRACTION.request_col_name}
237            WHERE act_req.odsobject = '{self._BW_EXTRACTION.odsobject}'
238                AND operation = 'A' AND status = '0'
239            )"""  # nosec: B608
240
241        return extraction_query

243    def _get_delta_query(self) -> Tuple[str, str]:
244        """Get a delta query for an SAP BW DSO.
245
246        An SAP BW DSO requires a join with a special type of table often called
247        activation requests table, in which BW tracks down the timestamps associated
248        with the several data loads that were performed into BW. Because the changelog
249        table only contains the active request id, and that cannot be sorted by the
250        downstream consumers to figure out the latest change, we need to join the
251        changelog table with this special table to get the activation requests
252        timestamps to then use them to figure out the latest changes in the delta load
253        logic afterwards.
254
255        Additionally, we also need to know which was the latest timestamp already loaded
256        into the lakehouse bronze layer. The latest timestamp should always be available
257        in the bronze dataset itself or in a dataset that tracks down all the actrequest
258        timestamps that were already loaded. So we get the max value out of the
259        respective actrequest timestamp column in that dataset.
260
261        Returns:
262            A query to submit to SAP BW for the delta data extraction. The query
263            is enclosed in parentheses so that Spark treats it as a table and supports
264            it in the dbtable option.
265        """
266        if not self._BW_EXTRACTION.min_timestamp:
267            from lakehouse_engine.io.reader_factory import ReaderFactory
268
269            latest_timestamp_data_df = ReaderFactory.get_data(
270                InputSpec(
271                    spec_id="data_with_latest_timestamp",
272                    data_format=self._BW_EXTRACTION.latest_timestamp_data_format,
273                    read_type=ReadType.BATCH.value,
274                    location=self._BW_EXTRACTION.latest_timestamp_data_location,
275                )
276            )
277            min_timestamp = latest_timestamp_data_df.transform(
278                Aggregators.get_max_value(
279                    self._BW_EXTRACTION.latest_timestamp_input_col
280                )
281            ).first()[0]
282        else:
283            min_timestamp = self._BW_EXTRACTION.min_timestamp
284
285        max_timestamp = (
286            self._BW_EXTRACTION.max_timestamp
287            if self._BW_EXTRACTION.max_timestamp
288            else self._get_max_timestamp(self._MAX_TIMESTAMP_QUERY)
289        )
290
291        if self._BW_EXTRACTION.act_req_join_condition:
292            join_condition = f"{self._BW_EXTRACTION.act_req_join_condition}"
293        else:
294            join_condition = (
295                f"changelog_tbl.request = "
296                f"act_req.{self._BW_EXTRACTION.request_col_name}"
297            )
298
299        base_query = f""" --# nosec
300        FROM {self._BW_EXTRACTION.changelog_table} AS changelog_tbl
301        JOIN {self._BW_EXTRACTION.act_request_table} AS act_req
302            ON {join_condition}
303        WHERE act_req.odsobject = '{self._BW_EXTRACTION.odsobject}'
304            AND act_req.timestamp > {min_timestamp}
305            AND act_req.timestamp <= {max_timestamp}
306            AND operation = 'A' AND status = '0')
307        """
308
309        main_cols = f"""
310            (SELECT changelog_tbl.*,
311                act_req.TIMESTAMP AS actrequest_timestamp,
312                CAST({self._BW_EXTRACTION.extraction_timestamp} AS DECIMAL(15,0))
313                    AS extraction_start_timestamp
314            """
315        # We join the main columns considered for the extraction with
316        # extra_cols_act_request that people might want to use, filtering to only
317        # add the comma and join the strings, in case extra_cols_act_request is
318        # not None or empty.
319        extraction_query_cols = ",".join(
320            filter(None, [main_cols, self._BW_EXTRACTION.extra_cols_act_request])
321        )
322
323        extraction_query = extraction_query_cols + base_query
324
325        predicates_query = f"""
326        (SELECT DISTINCT({self._BW_EXTRACTION.partition_column})
327        {base_query}
328        """
329
330        return extraction_query, predicates_query

244    def _get_delta_query(self) -> Tuple[str, str]:
245        """Get a delta query for an SAP BW DSO.
246
247        An SAP BW DSO requires a join with a special type of table often called
248        activation requests table, in which BW tracks down the timestamps associated
249        with the several data loads that were performed into BW. Because the changelog
250        table only contains the active request id, and that cannot be sorted by the
251        downstream consumers to figure out the latest change, we need to join the
252        changelog table with this special table to get the activation requests
253        timestamps to then use them to figure out the latest changes in the delta load
254        logic afterwards.
255
256        Additionally, we also need to know which was the latest timestamp already loaded
257        into the lakehouse bronze layer. The latest timestamp should always be available
258        in the bronze dataset itself or in a dataset that tracks down all the actrequest
259        timestamps that were already loaded. So we get the max value out of the
260        respective actrequest timestamp column in that dataset.
261
262        Returns:
263            A query to submit to SAP BW for the delta data extraction. The query
264            is enclosed in parentheses so that Spark treats it as a table and supports
265            it in the dbtable option.
266        """
267        if not self._BW_EXTRACTION.min_timestamp:
268            from lakehouse_engine.io.reader_factory import ReaderFactory
269
270            latest_timestamp_data_df = ReaderFactory.get_data(
271                InputSpec(
272                    spec_id="data_with_latest_timestamp",
273                    data_format=self._BW_EXTRACTION.latest_timestamp_data_format,
274                    read_type=ReadType.BATCH.value,
275                    location=self._BW_EXTRACTION.latest_timestamp_data_location,
276                )
277            )
278            min_timestamp = latest_timestamp_data_df.transform(
279                Aggregators.get_max_value(
280                    self._BW_EXTRACTION.latest_timestamp_input_col
281                )
282            ).first()[0]
283        else:
284            min_timestamp = self._BW_EXTRACTION.min_timestamp
285
286        max_timestamp = (
287            self._BW_EXTRACTION.max_timestamp
288            if self._BW_EXTRACTION.max_timestamp
289            else self._get_max_timestamp(self._MAX_TIMESTAMP_QUERY)
290        )
291
292        if self._BW_EXTRACTION.act_req_join_condition:
293            join_condition = f"{self._BW_EXTRACTION.act_req_join_condition}"
294        else:
295            join_condition = (
296                f"changelog_tbl.request = "
297                f"act_req.{self._BW_EXTRACTION.request_col_name}"
298            )
299
300        base_query = f""" --# nosec
301        FROM {self._BW_EXTRACTION.changelog_table} AS changelog_tbl
302        JOIN {self._BW_EXTRACTION.act_request_table} AS act_req
303            ON {join_condition}
304        WHERE act_req.odsobject = '{self._BW_EXTRACTION.odsobject}'
305            AND act_req.timestamp > {min_timestamp}
306            AND act_req.timestamp <= {max_timestamp}
307            AND operation = 'A' AND status = '0')
308        """
309
310        main_cols = f"""
311            (SELECT changelog_tbl.*,
312                act_req.TIMESTAMP AS actrequest_timestamp,
313                CAST({self._BW_EXTRACTION.extraction_timestamp} AS DECIMAL(15,0))
314                    AS extraction_start_timestamp
315            """
316        # We join the main columns considered for the extraction with
317        # extra_cols_act_request that people might want to use, filtering to only
318        # add the comma and join the strings, in case extra_cols_act_request is
319        # not None or empty.
320        extraction_query_cols = ",".join(
321            filter(None, [main_cols, self._BW_EXTRACTION.extra_cols_act_request])
322        )
323
324        extraction_query = extraction_query_cols + base_query
325
326        predicates_query = f"""
327        (SELECT DISTINCT({self._BW_EXTRACTION.partition_column})
328        {base_query}
329        """
330
331        return extraction_query, predicates_query
