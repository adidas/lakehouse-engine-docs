 21    @classmethod
 22    def store_engine_usage(
 23        cls,
 24        acon: dict,
 25        func_name: str,
 26        collect_engine_usage: str = None,
 27        spark_confs: dict = None,
 28    ) -> None:
 29        """Collects and store Lakehouse Engine usage statistics.
 30
 31        These statistics include the acon and other relevant information, such as
 32        the lakehouse engine version and the functions/algorithms being used.
 33
 34        Args:
 35            acon: acon dictionary file.
 36            func_name: function name that called this log acon.
 37            collect_engine_usage: Lakehouse usage statistics collection strategy.
 38            spark_confs: optional dictionary with the spark confs to be used when
 39                collecting the engine usage.
 40        """
 41        try:
 42            if (
 43                collect_engine_usage
 44                in [
 45                    CollectEngineUsage.ENABLED.value,
 46                    CollectEngineUsage.PROD_ONLY.value,
 47                ]
 48                or ExecEnv.ENGINE_CONFIG.collect_engine_usage
 49                in CollectEngineUsage.ENABLED.value
 50            ):
 51                start_timestamp = datetime.now()
 52                timestamp_str = start_timestamp.strftime("%Y%m%d%H%M%S")
 53
 54                usage_stats: Dict = {"acon": ConfigUtils.remove_sensitive_info(acon)}
 55                EngineUsageStats.get_spark_conf_values(usage_stats, spark_confs)
 56
 57                engine_usage_path = None
 58                if usage_stats["environment"] == "prod":
 59                    engine_usage_path = ExecEnv.ENGINE_CONFIG.engine_usage_path
 60                elif collect_engine_usage != CollectEngineUsage.PROD_ONLY.value:
 61                    engine_usage_path = ExecEnv.ENGINE_CONFIG.engine_dev_usage_path
 62
 63                if engine_usage_path is not None:
 64                    usage_stats["function"] = func_name
 65                    usage_stats["engine_version"] = ConfigUtils.get_engine_version()
 66                    usage_stats["start_timestamp"] = start_timestamp
 67                    usage_stats["year"] = start_timestamp.year
 68                    usage_stats["month"] = start_timestamp.month
 69                    run_id_extracted = re.search(
 70                        "run-([1-9]\\w+)", usage_stats["run_id"]
 71                    )
 72                    usage_stats["run_id"] = (
 73                        run_id_extracted.group(1) if run_id_extracted else ""
 74                    )
 75
 76                    log_file_name = f"eng_usage_{func_name}_{timestamp_str}.json"
 77
 78                    usage_stats_str = json.dumps(usage_stats, default=str)
 79
 80                    url = urlparse(
 81                        f"{engine_usage_path}/{usage_stats['dp_name']}/"
 82                        f"{start_timestamp.year}/{start_timestamp.month}/"
 83                        f"{log_file_name}",
 84                        allow_fragments=False,
 85                    )
 86
 87                    try:
 88                        FileStorageFunctions.write_payload(
 89                            engine_usage_path, url, usage_stats_str
 90                        )
 91                        cls._LOGGER.info("Storing Lakehouse Engine usage statistics")
 92                    except FileNotFoundError as e:
 93                        cls._LOGGER.error(
 94                            f"Could not write engine stats into file: {e}."
 95                        )
 96        except Exception as e:
 97            cls._LOGGER.error(
 98                "Failed while collecting the lakehouse engine stats: "
 99                f"Unexpected {e=}, {type(e)=}."
100            )

102    @classmethod
103    def get_spark_conf_values(cls, usage_stats: dict, spark_confs: dict) -> None:
104        """Get information from spark session configurations.
105
106        Args:
107            usage_stats: usage_stats dictionary file.
108            spark_confs: optional dictionary with the spark tags to be used when
109                collecting the engine usage.
110        """
111        spark_confs = (
112            EngineStats.DEF_SPARK_CONFS.value
113            if spark_confs is None
114            else EngineStats.DEF_SPARK_CONFS.value | spark_confs
115        )
116
117        for spark_conf_key, spark_conf_value in spark_confs.items():
118            # whenever the spark_conf_value has #, it means it is an array, so we need
119            # to split it and adequately process it
120            if "#" in spark_conf_value:
121                array_key = spark_conf_value.split("#")
122                array_values = ast.literal_eval(
123                    ExecEnv.SESSION.conf.get(array_key[0], "[]")
124                )
125                final_value = [
126                    key_val["value"]
127                    for key_val in array_values
128                    if key_val["key"] == array_key[1]
129                ]
130                usage_stats[spark_conf_key] = (
131                    final_value[0] if len(final_value) > 0 else ""
132                )
133            else:
134                usage_stats[spark_conf_key] = ExecEnv.SESSION.conf.get(
135                    spark_conf_value, ""
136                )
