31class FilterSensitiveData(logging.Filter):
32    """Logging filter to hide sensitive data from being shown in the logs."""
33
34    def filter(self, record: logging.LogRecord) -> bool:  # noqa: A003
35        """Hide sensitive information from being shown in the logs.
36
37        Based on the configured regex and replace strings, the content of the log
38        records is replaced and then all the records are allowed to be logged
39        (return True).
40
41        Args:
42            record: the LogRecord event being logged.
43
44        Returns:
45            The transformed record to be logged.
46        """
47        for key_reg in SENSITIVE_KEYS_REG:
48            record.msg = re.sub(key_reg["regex"], key_reg["replace"], str(record.msg))
49        return True

52class LoggingHandler(object):
53    """Handle the logging of the lakehouse engine project."""
54
55    def __init__(self, class_name: str):
56        """Construct a LoggingHandler instance.
57
58        Args:
59            class_name: name of the class to be indicated in the logs.
60        """
61        self._logger: logging.Logger = logging.getLogger(class_name)
62        self._logger.setLevel(logging.DEBUG)
63        self._logger.addFilter(FilterSensitiveData())
64        lsh = logging.StreamHandler()
65        lsh.setLevel(logging.DEBUG)
66        lsh.setFormatter(FORMATTER)
67        if not self._logger.hasHandlers():
68            # avoid keep adding handlers and therefore duplicate messages
69            self._logger.addHandler(lsh)
70
71    def get_logger(self) -> logging.Logger:
72        """Get the _logger instance variable.
73
74        :return logging.Logger: the logger object.
75        """
76        return self._logger
