123    def extract_columns_from_mapping(
124        cls,
125        columns: dict,
126        is_dimension: bool,
127        extract_column_without_alias: bool = False,
128        table_alias: Optional[str] = None,
129        is_extracted_value_as_name: bool = True,
130    ) -> Union[tuple[list[str], list[str]], list[str]]:
131        """Extract and transform columns to SQL select statement.
132
133        Args:
134            columns: data to extract the columns.
135            is_dimension: flag identifying if is a dimension or a metric.
136            extract_column_without_alias: flag to inform if it's to extract columns
137                without aliases.
138            table_alias: name or alias from the source table.
139            is_extracted_value_as_name: identify if the extracted value is the
140                column name.
141        """
142        column_with_alias = (
143            "".join([table_alias, ".", "{} as {}"]) if table_alias else "{} as {}"
144        )
145        column_without_alias = (
146            "".join([table_alias, ".", "{}"]) if table_alias else "{}"
147        )
148
149        extracted_columns_with_alias = []
150        extracted_columns_without_alias = []
151        for column_name, column_value in columns.items():
152            if extract_column_without_alias:
153                extracted_column_without_alias = column_without_alias.format(
154                    cls._get_column_format_without_alias(
155                        is_dimension,
156                        column_name,
157                        column_value,
158                        is_extracted_value_as_name,
159                    )
160                )
161                extracted_columns_without_alias.append(extracted_column_without_alias)
162
163            extracted_column_with_alias = column_with_alias.format(
164                *cls._extract_column_with_alias(
165                    is_dimension,
166                    column_name,
167                    column_value,
168                    is_extracted_value_as_name,
169                )
170            )
171            extracted_columns_with_alias.append(extracted_column_with_alias)
172
173        return (
174            (extracted_columns_with_alias, extracted_columns_without_alias)
175            if extract_column_without_alias
176            else extracted_columns_with_alias
177        )

233    @classmethod
234    def get_cadence_configuration_at_end_date(cls, end_date: datetime) -> dict:
235        """A dictionary that corresponds to the conclusion of a cadence.
236
237        Any end date inputted by the user we check this end date is actually end of
238            a cadence (YEAR, QUARTER, MONTH, WEEK).
239        If the user input is 2024-03-31 this is a month end and a quarter end that
240            means any use cases configured as month or quarter need to be calculated.
241
242        Args:
243            end_date: base end date.
244        """
245        init_end_date_dict = {}
246
247        expected_end_cadence_date = pendulum.datetime(
248            int(end_date.strftime("%Y")),
249            int(end_date.strftime("%m")),
250            int(end_date.strftime("%d")),
251        ).replace(tzinfo=None)
252
253        # Validating YEAR cadence
254        if end_date == expected_end_cadence_date.last_of("year"):
255            init_end_date_dict["YEAR"] = "N"
256
257        # Validating QUARTER cadence
258        if end_date == expected_end_cadence_date.last_of("quarter"):
259            init_end_date_dict["QUARTER"] = "N"
260
261        # Validating MONTH cadence
262        if end_date == datetime(
263            int(end_date.strftime("%Y")),
264            int(end_date.strftime("%m")),
265            calendar.monthrange(
266                int(end_date.strftime("%Y")), int(end_date.strftime("%m"))
267            )[1],
268        ):
269            init_end_date_dict["MONTH"] = "N"
270
271        # Validating WEEK cadence
272        if end_date == expected_end_cadence_date.end_of("week").replace(
273            hour=0, minute=0, second=0, microsecond=0
274        ):
275            init_end_date_dict["WEEK"] = "N"
276
277        init_end_date_dict["DAY"] = "N"
278
279        return init_end_date_dict

304    @classmethod
305    def _get_cadences_to_execute(
306        cls,
307        configured_cadences: dict,
308        cadence: str,
309        cadence_configuration_at_end_date: dict,
310        rerun_flag: str,
311    ) -> dict:
312        """Get cadences to execute.
313
314        Args:
315            cadence: cadence to process.
316            configured_cadences: configured use case reconciliation window.
317            cadence_configuration_at_end_date: cadences to execute at the end date.
318            rerun_flag: flag indicating if it's a rerun or a normal run.
319        """
320        cadences_to_execute = {}
321        cad_order = GABCadence.get_ordered_cadences()
322
323        for snapshot_cadence, snapshot_flag in configured_cadences.items():
324            if (
325                (cad_order[cadence] > cad_order[snapshot_cadence])
326                and (rerun_flag == "Y")
327            ) or snapshot_cadence in cadence_configuration_at_end_date:
328                cadences_to_execute[snapshot_cadence] = snapshot_flag
329            elif snapshot_cadence not in cadence_configuration_at_end_date:
330                continue
331
332        return cls._sort_cadences_to_execute(cadences_to_execute, cad_order)

334    @classmethod
335    def _sort_cadences_to_execute(
336        cls, cadences_to_execute: dict, cad_order: dict
337    ) -> dict:
338        """Sort the cadences to execute.
339
340        Args:
341            cadences_to_execute: cadences to execute.
342            cad_order: all cadences with order.
343        """
344        # ordering it because when grouping cadences with snapshot and without snapshot
345        # can impact the cadence ordering.
346        sorted_cadences_to_execute: dict = dict(
347            sorted(
348                cadences_to_execute.items(),
349                key=lambda item: cad_order.get(item[0]),  # type: ignore
350            )
351        )
352        # ordering cadences to execute it from bigger (YEAR) to smaller (DAY)
353        cadences_to_execute_items = []
354
355        for cadence_name, cadence_value in sorted_cadences_to_execute.items():
356            cadences_to_execute_items.append((cadence_name, cadence_value))
357
358        cadences_sorted_by_bigger_cadence_to_execute: dict = dict(
359            reversed(cadences_to_execute_items)
360        )
361
362        return cadences_sorted_by_bigger_cadence_to_execute

364    @classmethod
365    def _get_configured_cadences_by_snapshot(
366        cls,
367        cadence: str,
368        selected_reconciliation_window: dict,
369        cadence_configuration_at_end_date: dict,
370    ) -> dict:
371        """Get configured cadences to execute.
372
373        Args:
374            cadence: selected cadence.
375            selected_reconciliation_window: configured use case reconciliation window.
376            cadence_configuration_at_end_date: cadences to execute at the end date.
377
378        Returns:
379            Each cadence with the corresponding information if it's to execute with
380                snapshot or not.
381        """
382        cadences_by_snapshot = {}
383
384        (
385            no_snapshot_cadences,
386            snapshot_cadences,
387        ) = cls._generate_reconciliation_by_snapshot(
388            cadence, selected_reconciliation_window
389        )
390
391        for snapshot_cadence, snapshot_flag in no_snapshot_cadences.items():
392            if snapshot_cadence in cadence_configuration_at_end_date:
393                cadences_by_snapshot[snapshot_cadence] = snapshot_flag
394
395                cls._LOGGER.info(f"{snapshot_cadence} is present in {cadence} cadence")
396                break
397
398        cadences_by_snapshot.update(snapshot_cadences)
399
400        if (not cadences_by_snapshot) and (
401            cadence in cadence_configuration_at_end_date
402        ):
403            cadences_by_snapshot[cadence] = "N"
404
405        return cadences_by_snapshot

407    @classmethod
408    def _generate_reconciliation_by_snapshot(
409        cls, cadence: str, selected_reconciliation_window: dict
410    ) -> tuple[dict, dict]:
411        """Generate reconciliation by snapshot.
412
413        Args:
414            cadence: cadence to process.
415            selected_reconciliation_window: configured use case reconciliation window.
416        """
417        cadence_snapshot_configuration = {cadence: "N"}
418        for cadence in GABCadence.get_cadences():
419            cls._add_cadence_snapshot_to_cadence_snapshot_config(
420                cadence, selected_reconciliation_window, cadence_snapshot_configuration
421            )
422        cadence_snapshot_configuration = dict(
423            sorted(
424                cadence_snapshot_configuration.items(),
425                key=(
426                    lambda item: GABCadence.get_ordered_cadences().get(  # type: ignore
427                        item[0]
428                    )
429                ),
430            )
431        )
432
433        cadence_snapshot_configuration = dict(
434            reversed(list(cadence_snapshot_configuration.items()))
435        )
436
437        cadences_without_snapshot = {
438            key: value
439            for key, value in cadence_snapshot_configuration.items()
440            if value == "N"
441        }
442
443        cadences_with_snapshot = {
444            key: value
445            for key, value in cadence_snapshot_configuration.items()
446            if value == "Y"
447        }
448
449        return cadences_with_snapshot, cadences_without_snapshot

523    @classmethod
524    def _get_multiple_years_partition(
525        cls, start_date: str, end_date: str, years: list[str]
526    ) -> str:
527        """Return partition when executing multiple years (>1).
528
529        Args:
530            start_date: start of the period.
531            end_date: end of the period.
532            years: list of years.
533        """
534        start_date_month = cls._extract_date_part_from_date("MONTH", start_date)
535        start_date_day = cls._extract_date_part_from_date("DAY", start_date)
536
537        end_date_month = cls._extract_date_part_from_date("MONTH", end_date)
538        end_date_day = cls._extract_date_part_from_date("DAY", end_date)
539
540        year_statement = "(year = {0} and (".format(years[0]) + "{})"
541        if start_date_month != "12":
542            start_date_partition = year_statement.format(
543                "(month = {0} and day between {1} and 31)".format(
544                    start_date_month, start_date_day
545                )
546                + " or (month between {0} and 12)".format(int(start_date_month) + 1)
547            )
548        else:
549            start_date_partition = year_statement.format(
550                "month = {0} and day between {1} and 31".format(
551                    start_date_month, start_date_day
552                )
553            )
554
555        period_years_partition = ""
556
557        if len(years) == 3:
558            period_years_partition = ") or (year = {0}".format(years[1])
559        elif len(years) > 3:
560            period_years_partition = ") or (year between {0} and {1})".format(
561                years[1], years[-2]
562            )
563
564        if end_date_month != "01":
565            end_date_partition = (
566                ") or (year = {0} and ((month between 01 and {1})".format(
567                    years[-1], int(end_date_month) - 1
568                )
569                + " or (month = {0} and day between 1 and {1})))".format(
570                    end_date_month, end_date_day
571                )
572            )
573        else:
574            end_date_partition = (
575                ") or (year = {0} and month = 1 and day between 01 and {1})".format(
576                    years[-1], end_date_day
577                )
578            )
579        partition_condition = (
580            start_date_partition + period_years_partition + end_date_partition
581        )
582
583        return partition_condition

585    @classmethod
586    def _get_single_year_partition(cls, start_date: str, end_date: str) -> str:
587        """Return partition when executing a single year.
588
589        Args:
590            start_date: start of the period.
591            end_date: end of the period.
592        """
593        start_date_year = cls._extract_date_part_from_date("YEAR", start_date)
594        start_date_month = cls._extract_date_part_from_date("MONTH", start_date)
595        start_date_day = cls._extract_date_part_from_date("DAY", start_date)
596
597        end_date_year = cls._extract_date_part_from_date("YEAR", end_date)
598        end_date_month = cls._extract_date_part_from_date("MONTH", end_date)
599        end_date_day = cls._extract_date_part_from_date("DAY", end_date)
600
601        if start_date_month != end_date_month:
602            months = []
603            for i in range(int(start_date_month), int(end_date_month) + 1):
604                months.append(i)
605
606            start_date_partition = (
607                "year = {0} and ((month={1} and day between {2} and 31)".format(
608                    start_date_year, months[0], start_date_day
609                )
610            )
611            period_years_partition = ""
612            if len(months) == 2:
613                period_years_partition = start_date_partition
614            elif len(months) == 3:
615                period_years_partition = (
616                    start_date_partition + " or (month = {0})".format(months[1])
617                )
618            elif len(months) > 3:
619                period_years_partition = (
620                    start_date_partition
621                    + " or (month between {0} and {1})".format(months[1], months[-2])
622                )
623            partition_condition = (
624                period_years_partition
625                + " or (month = {0} and day between 1 and {1}))".format(
626                    end_date_month, end_date_day
627                )
628            )
629        else:
630            partition_condition = (
631                "year = {0} and month = {1} and day between {2} and {3}".format(
632                    end_date_year, end_date_month, start_date_day, end_date_day
633                )
634            )
635
636        return partition_condition

640        """Extract date part from string date.
641
642        Args:
643            part: date part (possible values: DAY, MONTH, YEAR)
644            date: string date.
645        """

167                    column_name,

169                    is_extracted_value_as_name,

236        """A dictionary that corresponds to the conclusion of a cadence.

253

256            init_end_date_dict["YEAR"] = "N"

258        # Validating QUARTER cadence
259        if end_date == expected_end_cadence_date.last_of("quarter"):
260            init_end_date_dict["QUARTER"] = "N"
261
262        # Validating MONTH cadence
263        if end_date == datetime(
264            int(end_date.strftime("%Y")),

266            calendar.monthrange(
267                int(end_date.strftime("%Y")), int(end_date.strftime("%m"))
268            )[1],
269        ):
270            init_end_date_dict["MONTH"] = "N"
271
272        # Validating WEEK cadence
273        if end_date == expected_end_cadence_date.end_of("week").replace(
274            hour=0, minute=0, second=0, microsecond=0
275        ):
276            init_end_date_dict["WEEK"] = "N"
277
278        init_end_date_dict["DAY"] = "N"
279
280        return init_end_date_dict

305    @classmethod
306    def _get_cadences_to_execute(
307        cls,
308        configured_cadences: dict,
309        cadence: str,
310        cadence_configuration_at_end_date: dict,
311        rerun_flag: str,
312    ) -> dict:
313        """Get cadences to execute.
314
315        Args:
316            cadence: cadence to process.
317            configured_cadences: configured use case reconciliation window.
318            cadence_configuration_at_end_date: cadences to execute at the end date.
319            rerun_flag: flag indicating if it's a rerun or a normal run.
320        """
321        cadences_to_execute = {}
322        cad_order = GABCadence.get_ordered_cadences()
323
324        for snapshot_cadence, snapshot_flag in configured_cadences.items():
325            if (
326                (cad_order[cadence] > cad_order[snapshot_cadence])
327                and (rerun_flag == "Y")
328            ) or snapshot_cadence in cadence_configuration_at_end_date:
329                cadences_to_execute[snapshot_cadence] = snapshot_flag
330            elif snapshot_cadence not in cadence_configuration_at_end_date:
331                continue
332
333        return cls._sort_cadences_to_execute(cadences_to_execute, cad_order)

335    @classmethod
336    def _sort_cadences_to_execute(
337        cls, cadences_to_execute: dict, cad_order: dict
338    ) -> dict:
339        """Sort the cadences to execute.
340
341        Args:
342            cadences_to_execute: cadences to execute.
343            cad_order: all cadences with order.
344        """
345        # ordering it because when grouping cadences with snapshot and without snapshot
346        # can impact the cadence ordering.
347        sorted_cadences_to_execute: dict = dict(
348            sorted(
349                cadences_to_execute.items(),
350                key=lambda item: cad_order.get(item[0]),  # type: ignore
351            )
352        )
353        # ordering cadences to execute it from bigger (YEAR) to smaller (DAY)
354        cadences_to_execute_items = []
355
356        for cadence_name, cadence_value in sorted_cadences_to_execute.items():
357            cadences_to_execute_items.append((cadence_name, cadence_value))
358
359        cadences_sorted_by_bigger_cadence_to_execute: dict = dict(
360            reversed(cadences_to_execute_items)
361        )
362
363        return cadences_sorted_by_bigger_cadence_to_execute

365    @classmethod
366    def _get_configured_cadences_by_snapshot(
367        cls,
368        cadence: str,
369        selected_reconciliation_window: dict,
370        cadence_configuration_at_end_date: dict,
371    ) -> dict:
372        """Get configured cadences to execute.
373
374        Args:
375            cadence: selected cadence.
376            selected_reconciliation_window: configured use case reconciliation window.
377            cadence_configuration_at_end_date: cadences to execute at the end date.
378
379        Returns:
380            Each cadence with the corresponding information if it's to execute with
381                snapshot or not.
382        """
383        cadences_by_snapshot = {}
384
385        (
386            no_snapshot_cadences,
387            snapshot_cadences,
388        ) = cls._generate_reconciliation_by_snapshot(
389            cadence, selected_reconciliation_window
390        )
391
392        for snapshot_cadence, snapshot_flag in no_snapshot_cadences.items():
393            if snapshot_cadence in cadence_configuration_at_end_date:
394                cadences_by_snapshot[snapshot_cadence] = snapshot_flag
395
396                cls._LOGGER.info(f"{snapshot_cadence} is present in {cadence} cadence")
397                break
398
399        cadences_by_snapshot.update(snapshot_cadences)
400
401        if (not cadences_by_snapshot) and (
402            cadence in cadence_configuration_at_end_date
403        ):
404            cadences_by_snapshot[cadence] = "N"
405
406        return cadences_by_snapshot

412        """Generate reconciliation by snapshot.
413
414        Args:
415            cadence: cadence to process.

427                    lambda item: GABCadence.get_ordered_cadences().get(  # type: ignore
428                        item[0]
429                    )
430                ),
431            )
432        )
433
434        cadence_snapshot_configuration = dict(
435            reversed(list(cadence_snapshot_configuration.items()))
436        )
437
438        cadences_without_snapshot = {
439            key: value
440            for key, value in cadence_snapshot_configuration.items()
441            if value == "N"
442        }
443
444        cadences_with_snapshot = {
445            key: value
446            for key, value in cadence_snapshot_configuration.items()
447            if value == "Y"
448        }
449
